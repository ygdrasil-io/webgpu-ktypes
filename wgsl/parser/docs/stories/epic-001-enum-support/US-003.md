### Code à ajouter dans TranslationUnit.kt

// =============================================================================
// ENUM DECLARATION SUPPORT
// =============================================================================

/**
 * A struct member declaration.
 */
data class StructMember(
    /** Attributes on this member. */
    val attributes: List<Attribute>,
    /** The name of the member. */
    val name: String,
    /** The type of the member. */
    val type: TypeDecl,
    /** The default value (if any). */
    val defaultValue: Expression?,
    val span: Span,
)

/**
 * An enumeration member.
 * 
 * Represents a single value in an enum declaration.
 * Each member can have an optional explicit value.
 */
data class EnumMember(
    /** Attributes on this enum member. */
    val attributes: List<Attribute>,
    /** The name of the enum member. */
    val name: String,
    /** The explicit value of the enum member (null if auto-assigned). */
    val value: Expression?,
    /** The span of this enum member. */
    val span: Span,
)

/**
 * An enumeration declaration.
 * 
 * Represents a WGSL enum type declaration like:
 * enum MyEnum {
 *     VALUE1,
 *     VALUE2 = 42,
 *     VALUE3
 * }
 */
data class EnumDecl(
    /** Attributes on this enum. */
    val attributes: List<Attribute>,
    /** The name of the enum. */
    val name: String,
    /** The members of the enum. */
    val members: List<EnumMember>,
    /** The span covering the entire declaration. */
    override val span: Span,
) : GlobalDecl()

// =============================================================================
// PREDECLARED ENUMERANTS (Forward declaration for future use)
// =============================================================================

/**
 * Predeclared enumerant values.
 * These are built-in enum-like values defined by WGSL specification.
 * 
 * Examples: AddressMode.clamp_to_edge, FilterMode.nearest
 */
sealed class PredeclaredEnumerant {
    abstract val category: String
    abstract val value: String
    abstract val displayName: String
}

/**
 * Address mode values for texture sampling.
 */
data class AddressMode(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "AddressMode"
    override val displayName: String = value
    
    companion object {
        val CLAMP_TO_EDGE = AddressMode("clamp_to_edge")
        val REPEAT = AddressMode("repeat")
        val MIRROR_REPEAT = AddressMode("mirror_repeat")
    }
}

/**
 * Filter mode values for texture sampling.
 */
data class FilterMode(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "FilterMode"
    override val displayName: String = value
    
    companion object {
        val NEAREST = FilterMode("nearest")
        val LINEAR = FilterMode("linear")
    }
}

/**
 * Mipmap filter mode values.
 */
data class MipmapFilterMode(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "MipmapFilterMode"
    override val displayName: String = value
    
    companion object {
        val NEAREST = MipmapFilterMode("nearest")
        val LINEAR = MipmapFilterMode("linear")
    }
}

## 🧪 Tests Attendus

```kotlin
// EnumAstTest.kt
import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldHaveSize
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.ir.Span

class EnumAstTest : FunSpec({
    
    context("EnumMember") {
        test("create simple enum member") {
            val span = Span(0u, 10u)
            val member = EnumMember(
                attributes = emptyList(),
                name = "RED",
                value = null,
                span = span
            )
            
            member.name shouldBe "RED"
            member.value shouldBe null
            member.attributes shouldHaveSize 0
        }
        
        test("create enum member with value") {
            val span = Span(0u, 20u)
            val valueExpr = IntLiteral("42", Span(10u, 12u))
            val member = EnumMember(
                attributes = emptyList(),
                name = "VALUE2",
                value = valueExpr,
                span = span
            )
            
            member.name shouldBe "VALUE2"
            member.value shouldBe valueExpr
        }
    }
    
    context("EnumDecl") {
        test("create simple enum declaration") {
            val span = Span(0u, 30u)
            val members = listOf(
                EnumMember(emptyList(), "RED", null, Span(5u, 8u)),
                EnumMember(emptyList(), "GREEN", null, Span(10u, 15u)),
                EnumMember(emptyList(), "BLUE", null, Span(17u, 21u))
            )
            
            val enumDecl = EnumDecl(
                attributes = emptyList(),
                name = "Color",
                members = members,
                span = span
            )
            
            enumDecl.name shouldBe "Color"
            enumDecl.members shouldHaveSize 3
            enumDecl.attributes shouldHaveSize 0
        }
        
        test("create enum with attributes") {
            val span = Span(0u, 40u)
            val packedAttr = Attribute("packed", emptyList(), Span(0u, 7u))
            
            val enumDecl = EnumDecl(
                attributes = listOf(packedAttr),
                name = "MyEnum",
                members = emptyList(),
                span = span
            )
            
            enumDecl.attributes shouldHaveSize 1
            enumDecl.attributes[0].name shouldBe "packed"
        }
    }
    
    context("Predeclared Enumerants") {
        test("AddressMode values") {
            AddressMode.CLAMP_TO_EDGE.value shouldBe "clamp_to_edge"
            AddressMode.REPEAT.value shouldBe "repeat"
            AddressMode.MIRROR_REPEAT.value shouldBe "mirror_repeat"
        }
        
        test("FilterMode values") {
            FilterMode.NEAREST.value shouldBe "nearest"
            FilterMode.LINEAR.value shouldBe "linear"
        }
    }
})
```
