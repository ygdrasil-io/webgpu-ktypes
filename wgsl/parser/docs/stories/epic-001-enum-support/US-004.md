// =============================================================================

/**
 * Represents an enumeration type in the type system.
 * 
 * Enum types can have named members with optional explicit values.
 */
data class EnumType(
    /** The declaration of this enum type. */
    val decl: EnumDecl,
    /** The span of this type reference. */
    override val span: Span,
) : Type() {
    /**
     * Gets a member by name.
     * Returns null if the member doesn't exist.
     */
    fun getMember(name: String): EnumMember? {
        return decl.members.find { it.name == name }
    }
    
    /**
     * Gets all members of this enum.
     */
    fun getMembers(): List<EnumMember> = decl.members
    
    /**
     * Checks if this enum has a member with the given name.
     */
    fun hasMember(name: String): Boolean = decl.members.any { it.name == name }
    
    override fun toString(): String = decl.name
}

// Dans TypeResolver class, ajouter une méthode pour résoudre les EnumDecl
private fun resolveEnumType(decl: EnumDecl, span: Span): Type {
    // Vérifier si on a déjà résolu ce type
    val cached = enumTypeCache[decl.name]
    if (cached != null) {
        return cached
    }
    
    // Créer un nouveau type enum
    val enumType = EnumType(decl, span)
    enumTypeCache[decl.name] = enumType
    
    // Indexer les membres pour un accès rapide
    decl.members.forEach { member ->
        enumMemberCache["${decl.name}.${member.name}"] = member
    }
    
    return enumType
}

// Mettre à jour la méthode resolveType pour gérer EnumDecl
private fun resolveType(decl: TypeDecl): Type? {
    return when (decl) {
        is NamedType -> resolveNamedType(decl)
        is EnumDecl -> resolveEnumType(decl, decl.span)
        // ... autres cas existants
    }
}

// Ajouter une méthode pour résoudre les member access sur les enum
private fun resolveEnumMemberAccess(expr: MemberAccessExpr): Expression {
    val baseType = resolveExpressionType(expr.base)
    
    if (baseType is EnumType) {
        val member = baseType.getMember(expr.member)
        if (member != null) {
            // Retourner une référence au membre
            return EnumMemberExpr(baseType, member, expr.span)
        } else {
            // Erreur: membre non trouvé
            addError("Enum '${baseType.decl.name}' has no member '${expr.member}'", expr.span)
        }
    }
    
    // Fallback au comportement existant
    return super.resolveMemberAccess(expr)
}

// Ajouter au cache de TypeResolver
private val enumTypeCache = mutableMapOf<String, EnumType>()
private val enumMemberCache = mutableMapOf<String, EnumMember>()

// Data class pour représenter une expression de membre d'enum
data class EnumMemberExpr(
    val enumType: EnumType,
    val member: EnumMember,
    override val span: Span,
) : Expression() {
    override fun <R> accept(visitor: ExpressionVisitor<R>): R {
        return visitor.visitEnumMember(this)
    }
}

// EnumTypeResolverTest.kt
import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldNotBe
import io.kotest.assertions.throwables.shouldThrow
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.parser.TypeResolver
import io.ygdrasil.wgsl.ir.Span

class EnumTypeResolverTest : FunSpec({
    
    lateinit var typeResolver: TypeResolver
    
    beforeTest {
        // Setup: créer un TypeResolver avec des déclarations d'enum
        val enumDecl = EnumDecl(
            attributes = emptyList(),
            name = "Color",
            members = listOf(
                EnumMember(emptyList(), "RED", null, Span(0u, 3u)),
                EnumMember(emptyList(), "GREEN", null, Span(5u, 10u)),
                EnumMember(emptyList(), "BLUE", IntLiteral("42", Span(12u, 14u)), Span(12u, 18u))
            ),
            span = Span(0u, 20u)
        )
        
        typeResolver = TypeResolver(listOf(enumDecl))
    }
    
    context("Enum Type Resolution") {
        test("resolve enum declaration to EnumType") {
            val enumType = typeResolver.resolveType("Color") as? EnumType
            
            enumType shouldNotBe null
            enumType?.decl?.name shouldBe "Color"
        }
        
        test("access enum members") {
            val enumType = typeResolver.resolveType("Color") as EnumType
            
            enumType.hasMember("RED") shouldBe true
            enumType.hasMember("GREEN") shouldBe true
            enumType.hasMember("BLUE") shouldBe true
            enumType.hasMember("INVALID") shouldBe false
        }
        
        test("get enum member by name") {
            val enumType = typeResolver.resolveType("Color") as EnumType
            
            val redMember = enumType.getMember("RED")
            redMember shouldNotBe null
            redMember?.name shouldBe "RED"
            
            val blueMember = enumType.getMember("BLUE")
            blueMember shouldNotBe null
            blueMember?.value shouldNotBe null
        }
        
        test("enum type can be used as variable type") {
            // Simuler: let color: Color = Color.RED;
            // Cela devrait réussir après l'intégration complète
        }
    }
    
    context("Enum Member Access") {
        test("resolve member access expression") {
            // Créer une expression: Color.RED
            val enumType = typeResolver.resolveType("Color") as EnumType
            val redMember = enumType.getMember("RED")!!
            
            val memberExpr = EnumMemberExpr(enumType, redMember, Span(0u, 10u))
            
            val resolvedType = typeResolver.resolveExpressionType(memberExpr)
            // Le type de Color.RED devrait être Color
            resolvedType shouldBe enumType
        }
    }
})
