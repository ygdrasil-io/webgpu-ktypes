### 1. LexerEnumTest.kt

```kotlin
package io.ygdrasil.wgsl.lexer

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.collections.shouldContain
import io.kotest.matchers.collections.shouldHaveSize
import io.kotest.matchers.shouldBe

/**
 * Tests for lexing enum-related tokens.
 */
class LexerEnumTest : FunSpec({
    
    context("enum keyword") {
        test("recognizes enum as keyword") {
            val tokens = Lexer("enum").tokenizeSignificant()
            tokens shouldHaveSize 1
            tokens[0].kind shouldBe TokenKind.ENUM
        }
        
        test("enum is a keyword") {
            TokenKind.ENUM.isKeyword shouldBe true
        }
        
        test("enum keyword in context") {
            val tokens = Lexer("enum MyEnum {}").tokenizeSignificant()
            tokens shouldContain TokenKind.ENUM
            tokens shouldContain TokenKind.IDENTIFIER
        }
    }
    
    context("enum declarations") {
        test("simple enum declaration") {
            val source = "enum Color { RED, GREEN, BLUE }"
            val tokens = Lexer(source).tokenizeSignificant()
            
            tokens shouldContain TokenKind.ENUM
            tokens.filter { it.kind == TokenKind.IDENTIFIER } shouldHaveSize 4 // Color, RED, GREEN, BLUE
            tokens shouldContain TokenKind.LEFT_BRACE
            tokens shouldContain TokenKind.RIGHT_BRACE
        }
        
        test("enum with explicit values") {
            val source = "enum Status { ACTIVE = 1, INACTIVE = 0 }"
            val tokens = Lexer(source).tokenizeSignificant()
            
            tokens shouldContain TokenKind.ENUM
            tokens shouldContain TokenKind.ASSIGN
            tokens.filter { it.kind == TokenKind.INT_LITERAL } shouldHaveSize 2
        }
        
        test("enum with attributes") {
            val source = "@packed enum MyEnum { VALUE }"
            val tokens = Lexer(source).tokenizeSignificant()
            
            tokens shouldContain TokenKind.AT
            tokens shouldContain TokenKind.ENUM
        }
    }
    
    context("enum member access") {
        test("member access syntax") {
            val source = "Color.RED"
            val tokens = Lexer(source).tokenizeSignificant()
            
            tokens shouldHaveSize 3
            tokens[0].kind shouldBe TokenKind.IDENTIFIER
            tokens[0].text shouldBe "Color"
            tokens[1].kind shouldBe TokenKind.DOT
            tokens[2].kind shouldBe TokenKind.IDENTIFIER
            tokens[2].text shouldBe "RED"
        }
    }
})
```

```kotlin
package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.collections.shouldHaveSize
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldNotBe
import io.kotest.matchers.types.shouldBeInstanceOf
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.lexer.Lexer

/**
 * Tests for parsing enum declarations.
 */
class EnumParserTest : FunSpec({
    
    context("simple enum") {
        test("parse simple enum declaration") {
            val source = "enum Color { RED, GREEN, BLUE }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            unit.declarations shouldHaveSize 1
            unit.declarations[0] shouldBeInstanceOf<EnumDecl>()
            
            val enumDecl = unit.declarations[0] as EnumDecl
            enumDecl.name shouldBe "Color"
            enumDecl.members shouldHaveSize 3
        }
        
        test("parse enum member names") {
            val source = "enum Color { RED, GREEN, BLUE }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val enumDecl = unit.declarations[0] as EnumDecl
            enumDecl.members.map { it.name } shouldBe listOf("RED", "GREEN", "BLUE")
        }
    }
    
    context("enum with explicit values") {
        test("parse enum with integer values") {
            val source = "enum Status { ACTIVE = 1, INACTIVE = 0 }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val enumDecl = unit.declarations[0] as EnumDecl
            enumDecl.members shouldHaveSize 2
            
            enumDecl.members[0].name shouldBe "ACTIVE"
            enumDecl.members[0].value shouldNotBe null
            enumDecl.members[0].value shouldBeInstanceOf<IntLiteral>()
            
            enumDecl.members[1].name shouldBe "INACTIVE"
            enumDecl.members[1].value shouldNotBe null
        }
        
        test("parse enum with mixed values") {
            val source = "enum Mixed { A, B = 42, C }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val enumDecl = unit.declarations[0] as EnumDecl
            enumDecl.members shouldHaveSize 3
            
            enumDecl.members[0].value shouldBe null
            enumDecl.members[1].value shouldNotBe null
            enumDecl.members[2].value shouldBe null
        }
    }
    
    context("enum with attributes") {
        test("parse enum with attribute") {
            val source = "@packed enum MyEnum { VALUE }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val enumDecl = unit.declarations[0] as EnumDecl
            enumDecl.attributes shouldHaveSize 1
            enumDecl.attributes[0].name shouldBe "packed"
        }
        
        test("parse enum member with attribute") {
            val source = "enum MyEnum { @size(4) VALUE }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val enumDecl = unit.declarations[0] as EnumDecl
            enumDecl.members shouldHaveSize 1
            enumDecl.members[0].attributes shouldHaveSize 1
            enumDecl.members[0].attributes[0].name shouldBe "size"
        }
    }
    
    context("trailing comma") {
        test("parse enum with trailing comma") {
            val source = "enum Direction { NORTH, SOUTH, }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val enumDecl = unit.declarations[0] as EnumDecl
            enumDecl.members shouldHaveSize 2
        }
    }
    
    context("multiple enums") {
        test("parse multiple enum declarations") {
            val source = """
                enum Color { RED, GREEN }
                enum Size { SMALL, LARGE }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            unit.declarations shouldHaveSize 2
            unit.declarations[0] shouldBeInstanceOf<EnumDecl>()
            unit.declarations[1] shouldBeInstanceOf<EnumDecl>()
        }
    }
    
    context("error handling") {
        test("error on missing brace") {
            val source = "enum Color { RED, GREEN, BLUE"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            // Devrait avoir des erreurs
            parser.errors.isNotEmpty() shouldBe true
        }
        
        test("error on duplicate member") {
            val source = "enum Color { RED, RED }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            // Devrait avoir des erreurs ou gérer les doublons
        }
    }
})
```

### 3. EnumAstTest.kt

```kotlin
package io.ygdrasil.wgsl.ast

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldHaveSize
import io.ygdrasil.wgsl.ir.Span

/**
 * Tests for enum AST classes.
 */
class EnumAstTest : FunSpec({
    
    context("EnumMember") {
        test("create simple enum member") {
            val span = Span(0u, 10u)
            val member = EnumMember(
                attributes = emptyList(),
                name = "RED",
                value = null,
                span = span
            )
            
            member.name shouldBe "RED"
            member.value shouldBe null
            member.attributes shouldHaveSize 0
        }
        
        test("create enum member with value") {
            val span = Span(0u, 20u)
            val valueExpr = IntLiteral("42", Span(10u, 12u))
            val member = EnumMember(
                attributes = emptyList(),
                name = "VALUE2",
                value = valueExpr,
                span = span
            )
            
            member.name shouldBe "VALUE2"
            member.value shouldBe valueExpr
        }
        
        test("create enum member with attributes") {
            val span = Span(0u, 25u)
            val attr = Attribute("size", emptyList(), Span(0u, 5u))
            val member = EnumMember(
                attributes = listOf(attr),
                name = "VALUE",
                value = null,
                span = span
            )
            
            member.attributes shouldHaveSize 1
            member.attributes[0].name shouldBe "size"
        }
    }
    
    context("EnumDecl") {
        test("create simple enum declaration") {
            val span = Span(0u, 30u)
            val members = listOf(
                EnumMember(emptyList(), "RED", null, Span(5u, 8u)),
                EnumMember(emptyList(), "GREEN", null, Span(10u, 15u)),
                EnumMember(emptyList(), "BLUE", null, Span(17u, 21u))
            )
            
            val enumDecl = EnumDecl(
                attributes = emptyList(),
                name = "Color",
                members = members,
                span = span
            )
            
            enumDecl.name shouldBe "Color"
            enumDecl.members shouldHaveSize 3
            enumDecl.attributes shouldHaveSize 0
        }
        
        test("create enum with attributes") {
            val span = Span(0u, 40u)
            val packedAttr = Attribute("packed", emptyList(), Span(0u, 7u))
            
            val enumDecl = EnumDecl(
                attributes = listOf(packedAttr),
                name = "MyEnum",
                members = emptyList(),
                span = span
            )
            
            enumDecl.attributes shouldHaveSize 1
            enumDecl.attributes[0].name shouldBe "packed"
        }
    }
})
```

### 4. EnumTypeResolverTest.kt

```kotlin
package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldNotBe
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.ir.Span

/**
 * Tests for enum type resolution.
 */
class EnumTypeResolverTest : FunSpec({
    
    lateinit var typeResolver: TypeResolver
    
    beforeTest {
        val enumDecl = EnumDecl(
            attributes = emptyList(),
            name = "Color",
            members = listOf(
                EnumMember(emptyList(), "RED", null, Span(0u, 3u)),
                EnumMember(emptyList(), "GREEN", null, Span(5u, 10u)),
                EnumMember(emptyList(), "BLUE", IntLiteral("42", Span(12u, 14u)), Span(12u, 18u))
            ),
            span = Span(0u, 20u)
        )
        
        typeResolver = TypeResolver(listOf(enumDecl))
    }
    
    context("Enum Type Resolution") {
        test("resolve enum declaration to EnumType") {
            val enumType = typeResolver.resolveType("Color") as? EnumType
            
            enumType shouldNotBe null
            enumType?.decl?.name shouldBe "Color"
        }
        
        test("access enum members") {
            val enumType = typeResolver.resolveType("Color") as EnumType
            
            enumType.hasMember("RED") shouldBe true
            enumType.hasMember("GREEN") shouldBe true
            enumType.hasMember("BLUE") shouldBe true
            enumType.hasMember("INVALID") shouldBe false
        }
        
        test("get enum member by name") {
            val enumType = typeResolver.resolveType("Color") as EnumType
            
            val redMember = enumType.getMember("RED")
            redMember shouldNotBe null
            redMember?.name shouldBe "RED"
            
            val blueMember = enumType.getMember("BLUE")
            blueMember shouldNotBe null
            blueMember?.value shouldNotBe null
        }
        
        test("get all enum members") {
            val enumType = typeResolver.resolveType("Color") as EnumType
            val members = enumType.getMembers()
            
            members shouldHaveSize 3
            members.map { it.name } shouldBe listOf("RED", "GREEN", "BLUE")
        }
    }
})
```

### 5. EnumIntegrationTest.kt

```kotlin
package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldNotBe
import io.kotest.matchers.types.shouldBeInstanceOf
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.lexer.Lexer

/**
 * Integration tests for enum support.
 */
class EnumIntegrationTest : FunSpec({
    
    context("enum in variable declarations") {
        test("use enum type for variable") {
            val source = """
                enum Color { RED, GREEN }
                let myColor: Color = Color.RED;
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            unit.declarations shouldHaveSize 2
            unit.declarations[0] shouldBeInstanceOf<EnumDecl>()
            unit.declarations[1] shouldBeInstanceOf<VariableDeclStatement>()
        }
    }
    
    context("enum in function parameters") {
        test("use enum type for function parameter") {
            val source = """
                enum Status { ACTIVE, INACTIVE }
                fn process(status: Status) { }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            unit.declarations shouldHaveSize 2
            unit.declarations[0] shouldBeInstanceOf<EnumDecl>()
            unit.declarations[1] shouldBeInstanceOf<FunctionDecl>()
        }
    }
    
    context("enum in function return type") {
        test("use enum type as return type") {
            val source = """
                enum Result { SUCCESS, FAILURE }
                fn getResult() -> Result { return Result.SUCCESS; }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            unit.declarations shouldHaveSize 2
        }
    }
    
    context("enum in struct") {
        test("use enum type in struct") {
            val source = """
                enum Mode { FAST, SLOW }
                struct Config {
                    mode: Mode,
                }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            unit.declarations shouldHaveSize 2
            unit.declarations[0] shouldBeInstanceOf<EnumDecl>()
            unit.declarations[1] shouldBeInstanceOf<StructDecl>()
        }
    }
    
    context("complex enum usage") {
        test("complete enum usage example") {
            val source = """
                enum TextureMode {
                    REPEAT,
                    MIRROR,
                    CLAMP = 2
                }
                
                @group(0) @binding(0)
                var<uniform> config: Config;
                
                struct Config {
                    mode: TextureMode,
                }
                
                @fragment
                fn main() -> @location(0) vec4<f32> {
                    let mode: TextureMode = TextureMode.REPEAT;
                    if (mode == TextureMode.REPEAT) {
                        return vec4<f32>(1.0);
                    }
                    return vec4<f32>(0.0);
                }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            // Devrait parser sans erreurs
            parser.errors.isEmpty() shouldBe true
            unit.declarations.isNotEmpty() shouldBe true
        }
    }
})
```
