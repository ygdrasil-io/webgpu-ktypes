## 📋 Description

**En tant que** : Développeur utilisant le parser WGSL  
**Je veux** : Que le parser puisse analyser la syntaxe `enum Name { value1, value2 = 10, }`  
**Afin de** : Construire l'AST approprié pour les déclarations d'enum

---

## ✅ Critères d'Acceptation

- [ ] Le parser reconnaît `enum Name { ... }` comme une déclaration valide
- [ ] Les membres de l'enum avec et sans valeurs explicites sont supportés
- [ ] La virgule finale optionnelle est supportée
- [ ] Les attributes sur les enum et leurs membres sont supportés
- [ ] Test: `Parser("enum Color { RED, GREEN, BLUE = 3, }").parse()` réussit
- [ ] Test: `Parser("enum Status { ACTIVE = 1, INACTIVE }").parse()` réussit
- [ ] Test: `Parser("@packed enum MyEnum { A, B }").parse()` réussit

## 🎯 Exemple de Code

### Syntax WGSL (à supporter)
```wgsl
// Enum simple
enum Color {
    RED,
    GREEN,
    BLUE
}

// Enum avec valeurs
enum Status {
    ACTIVE = 1,
    INACTIVE = 0,
    PENDING
}

// Enum avec virgule finale
enum Direction {
    NORTH,
    SOUTH,
    EAST,
    WEST,
}

// Enum avec attributes
@packed
enum MyEnum {
    @size(4) VALUE1,
    VALUE2 = 42
}
```

```kotlin
// Ajouter à GlobalDecl sealed class
sealed class GlobalDecl {
    // ... existant
    data class EnumDecl(
        val attributes: List<Attribute>,
        val name: String,
        val members: List<EnumMember>,
        override val span: Span,
    ) : GlobalDecl()
}

// Nouvelle data class
 data class EnumMember(
    val attributes: List<Attribute>,
    val name: String,
    val value: Expression?,  // null si pas de valeur explicite
    val span: Span,
)
```

### Modifications Parser attendues
```kotlin
// parser/Parser.kt

private fun parseEnumDeclaration(): EnumDecl {
    expect(TokenKind.ENUM)
    val start = currentToken.span.start
    
    val attributes = parseAttributes()
    val name = expectIdentifier()
    expect(TokenKind.LEFT_BRACE)
    
    val members = mutableListOf<EnumMember>()
    while (peek().kind != TokenKind.RIGHT_BRACE) {
        val memberAttrs = parseAttributes()
        val memberName = expectIdentifier()
        val memberValue = if (peek().kind == TokenKind.ASSIGN) {
            advance()
            parseExpression()
        } else {
            null
        }
        members.add(EnumMember(memberAttrs, memberName, memberValue, span))
        
        // Gérer la virgule optionnelle
        if (peek().kind == TokenKind.COMMA) {
            advance()
        }
    }
    expect(TokenKind.RIGHT_BRACE)
    
    return EnumDecl(attributes, name, members, spanFrom(start))
}
```

## 🧪 Tests Attendus

```kotlin
// EnumParserTest.kt
class EnumParserTest : FunSpec({
    test("parse simple enum") {
        val source = "enum Color { RED, GREEN, BLUE }"
        val parser = Parser(Lexer(source))
        val unit = parser.parse()
        
        unit.declarations shouldHaveSize 1
        val enumDecl = unit.declarations[0] as EnumDecl
        enumDecl.name shouldBe "Color"
        enumDecl.members shouldHaveSize 3
        enumDecl.members.map { it.name } shouldContainAll listOf("RED", "GREEN", "BLUE")
    }
    
    test("parse enum with explicit values") {
        val source = "enum Status { ACTIVE = 1, INACTIVE = 0 }"
        val parser = Parser(Lexer(source))
        val unit = parser.parse()
        
        val enumDecl = unit.declarations[0] as EnumDecl
        enumDecl.members[0].value shouldNotBe null
        enumDecl.members[1].value shouldNotBe null
    }
    
    test("parse enum with trailing comma") {
        val source = "enum Direction { NORTH, SOUTH, }"
        val parser = Parser(Lexer(source))
        val unit = parser.parse()
        
        val enumDecl = unit.declarations[0] as EnumDecl
        enumDecl.members shouldHaveSize 2
    }
    
    test("parse enum with attributes") {
        val source = "@packed enum MyEnum { @size(4) VALUE }"
        val parser = Parser(Lexer(source))
        val unit = parser.parse()
        
        val enumDecl = unit.declarations[0] as EnumDecl
        enumDecl.attributes shouldHaveSize 1
        enumDecl.members[0].attributes shouldHaveSize 1
    }
})
```
