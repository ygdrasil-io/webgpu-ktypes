// Lexer.kt - Fonctions de lexing des littéraux

/**
 * Lexes a decimal integer literal (e.g., 42, 42u, 42i, 42h).
 * 
 * WGSL standard suffixes:
 * - No suffix: signed integer (i32)
 * - i: signed integer (i32)
 * - u: unsigned integer (u32)
 * - f: floating point (f32)
 * - h: half-precision floating point (f16)
 * 
 * Non-standard suffixes (REMOVED):
 * - li, lu, lf (long integer/float - not in WGSL spec)
 */
private fun lexDecimalLiteral(): Token {
    val start = currentPosition
    
    // Consume digits
    while (peek().isDigit()) {
        advance()
    }
    
    // Check for suffix
    val suffix = peek().takeIf { it.isLetter() }?.let { char ->
        advance()
        // Check for two-character suffixes
        if (peek().isLetter()) {
            val secondChar = peek()
            val twoCharSuffix = char.toString() + secondChar
            
            // REMOVED: No longer accept non-standard suffixes
            when (twoCharSuffix) {
                // These were non-standard and have been removed
                "lf", "li", "lu" -> {
                    // Generate error for non-standard suffix
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid literal suffix '$twoCharSuffix'. " +
                                    "Valid suffixes are: i, u, f, h. " +
                                    "See WGSL specification for more details."
                        )
                    )
                    // Return the literal without the suffix for error recovery
                    Token(DECIMAL_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
                else -> {
                    // Unknown two-character suffix - error
                    advance() // consume second char
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid literal suffix '$twoCharSuffix'. " +
                                    "Valid suffixes are: i, u, f, h."
                        )
                    )
                    Token(DECIMAL_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition))
                }
            }
        } else {
            // Single character suffix
            when (char) {
                'i', 'u', 'f', 'h' -> {
                    // Valid standard suffixes
                    val suffixText = textFrom(start + (currentPosition - start - 1), currentPosition)
                    Token(DECIMAL_INTEGER_LITERAL, spanFrom(start), suffixText)
                }
                'l' -> {
                    // 'l' by itself was not standard in WGSL
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid literal suffix 'l'. " +
                                    "Valid suffixes are: i, u, f, h."
                        )
                    )
                    Token(DECIMAL_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
                else -> {
                    // Unknown suffix - error
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid literal suffix '$char'. " +
                                    "Valid suffixes are: i, u, f, h."
                        )
                    )
                    Token(DECIMAL_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
            }
        }
    } ?: run {
        // No suffix - default to signed integer
        Token(DECIMAL_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition))
    }
    
    return token
}

/**
 * Lexes a hexadecimal integer literal (e.g., 0x2A, 0x2Au, 0x2Ai, 0x2Ah).
 * 
 * WGSL standard suffixes:
 * - No suffix: signed integer (i32)
 * - i: signed integer (i32)
 * - u: unsigned integer (u32)
 * - f: floating point (f32)
 * - h: half-precision floating point (f16)
 */
private fun lexHexLiteral(): Token {
    val start = currentPosition
    
    // Expect 0x or 0X
    expect('0')
    val hexPrefix = if (peek() == 'x' || peek() == 'X') {
        advance()
        "0x"
    } else {
        errors.add(
            ErrorMessage(
                span = spanFrom(start),
                message = "Invalid hexadecimal literal format. Expected '0x' or '0X'."
            )
        )
        return Token(DECIMAL_INTEGER_LITERAL, spanFrom(start), "0")
    }
    
    // Consume hex digits
    while (peek().isHexDigit()) {
        advance()
    }
    
    // Check for suffix
    val suffix = peek().takeIf { it.isLetter() }?.let { char ->
        advance()
        // Check for two-character suffixes
        if (peek().isLetter()) {
            val secondChar = peek()
            val twoCharSuffix = char.toString() + secondChar
            
            when (twoCharSuffix) {
                "lf", "li", "lu" -> {
                    // Generate error for non-standard suffix
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid literal suffix '$twoCharSuffix'. " +
                                    "Valid suffixes are: i, u, f, h."
                        )
                    )
                    Token(HEX_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
                else -> {
                    advance()
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid literal suffix '$twoCharSuffix'. " +
                                    "Valid suffixes are: i, u, f, h."
                        )
                    )
                    Token(HEX_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition))
                }
            }
        } else {
            when (char) {
                'i', 'u', 'f', 'h' -> {
                    // Valid standard suffixes
                    Token(HEX_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition))
                }
                'l' -> {
                    errors.add(

                            message = "Invalid literal suffix 'l'. " +
                                    "Valid suffixes are: i, u, f, h."
                        )

                }
                else -> {
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid literal suffix '$char'. " +
                                    "Valid suffixes are: i, u, f, h."
                        )
                    )
                    Token(HEX_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
            }
        }
    } ?: run {
        Token(HEX_INTEGER_LITERAL, spanFrom(start), textFrom(start, currentPosition))
    }
    
    return token
}

/**
 * Lexes a floating point literal (e.g., 3.14, 3.14f, 3.14h).
 * 
 * Note: WGSL does not support decimal floating point literals without suffix.
 * All floating point literals MUST have a suffix (f or h).
 */
private fun lexFloatLiteral(): Token {
    val start = currentPosition
    
    // Consume integer part
    while (peek().isDigit()) {
        advance()
    }
    
    // Expect decimal point
    expect('.')
    
    // Consume fractional part
    while (peek().isDigit()) {
        advance()
    }
    
    // Floating point literals MUST have a suffix
    val suffix = peek().takeIf { it.isLetter() }?.let { char ->
        advance()
        // Check for two-character suffixes
        if (peek().isLetter()) {
            val secondChar = peek()
            val twoCharSuffix = char.toString() + secondChar
            
            when (twoCharSuffix) {
                "lf", "li", "lu" -> {
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid floating point literal suffix '$twoCharSuffix'. " +
                                    "Valid suffixes are: f, h."
                        )
                    )
                    Token(FLOAT_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
                else -> {
                    advance()
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid floating point literal suffix '$twoCharSuffix'. " +
                                    "Valid suffixes are: f, h."
                        )
                    )
                    Token(FLOAT_LITERAL, spanFrom(start), textFrom(start, currentPosition))
                }
            }
        } else {
            when (char) {
                'f', 'h' -> {
                    // Valid standard suffixes
                    Token(FLOAT_LITERAL, spanFrom(start), textFrom(start, currentPosition))
                }
                'l' -> {
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid floating point literal suffix 'l'. " +
                                    "Valid suffixes are: f, h."
                        )
                    )
                    Token(FLOAT_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
                else -> {
                    errors.add(
                        ErrorMessage(
                            span = spanFrom(start),
                            message = "Invalid floating point literal suffix '$char'. " +
                                    "Valid suffixes are: f, h."
                        )
                    )
                    Token(FLOAT_LITERAL, spanFrom(start), textFrom(start, currentPosition - 1))
                }
            }
        }
    } ?: run {
        // No suffix for float literal - ERROR in WGSL
        errors.add(
            ErrorMessage(
                span = spanFrom(start),
                message = "Floating point literals must have a suffix. " +
                        "Valid suffixes are: f, h."
            )
        )
        Token(FLOAT_LITERAL, spanFrom(start), textFrom(start, currentPosition))
    }
    
    return token
}

// Lexer.kt - Classe pour les messages d'erreur

/**
 * Represents a lexing error with position information.
 */
data class ErrorMessage(
    val span: Span,
    val message: String
)

/**
 * Helper function to create error messages for invalid suffixes.
 */
private fun invalidSuffixError(suffix: String, validSuffixes: List<String>, span: Span): ErrorMessage {
    return ErrorMessage(
        span = span,
        message = "Invalid literal suffix '$suffix'. " +
                "Valid WGSL suffixes are: ${validSuffixes.joinToString(", ")}. " +
                "See WGSL specification Section 3.2 (Lexical Conventions) for more details."
    )
}

// Constantes pour les suffixes valides
private val VALID_INTEGER_SUFFIXES = listOf("i", "u", "f", "h", "")
private val VALID_FLOAT_SUFFIXES = listOf("f", "h")
