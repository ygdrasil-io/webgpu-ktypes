### 3. ConformanceReporter.kt - Génération de rapports

```kotlin
// ConformanceReporter.kt
package io.ygdrasil.wgsl.parser.conformance

import io.ygdrasil.wgsl.lexer.Lexer
import io.ygdrasil.wgsl.parser.Parser

/**
 * Generates conformance reports based on test results.
 */
object ConformanceReporter {
    
    private val results = mutableMapOf<String, ConformanceResult>()
    
    /**
     * Represents the result of a conformance test.
     */
    data class ConformanceResult(
        val section: String,
        val subsection: String,
        val testName: String,
        val passed: Boolean,
        val errors: List<String> = emptyList(),
        val wgslSource: String? = null
    )
    
    /**
     * Records a conformance test result.
     */
    fun recordResult(result: ConformanceResult) {
        val key = "${result.section}.${result.subsection}.${result.testName}"
        results[key] = result
    }
    
    /**
     * Tests if a WGSL source code parses and validates successfully.
     */
    fun testConformance(
        section: String,
        subsection: String,
        testName: String,
        source: String,
        shouldPass: Boolean = true
    ): Boolean {
        val parser = Parser(Lexer(source))
        val unit = parser.parseAndValidate()
        
        val passed = if (shouldPass) {
            parser.errors.isEmpty()
        } else {
            parser.errors.isNotEmpty()
        }
        
        val result = ConformanceResult(
            section = section,
            subsection = subsection,
            testName = testName,
            passed = passed,
            errors = parser.errors,
            wgslSource = source
        )
        
        recordResult(result)
        return passed
    }
    
    /**
     * Generates a formatted conformance report.
     */
    fun generateReport(): String {
        val builder = StringBuilder()
        
        // Header
        builder.appendLine("# WGSL Parser Conformance Report")
        builder.appendLine()
        builder.appendLine("**Generated:** ${java.time.LocalDateTime.now()}")
        builder.appendLine("**Specification:** WGSL CRD")
        builder.appendLine("**Parser Version:** ${getVersion()}")
        builder.appendLine()
        
        // Statistics
        val totalTests = results.size
        val passedTests = results.values.count { it.passed }
        val failedTests = totalTests - passedTests
        val conformancePercentage = (passedTests.toDouble() / totalTests * 100).toInt()
        
        builder.appendLine("## Overall Conformance: **$conformancePercentage%**")
        builder.appendLine()
        builder.appendLine("| Metric | Value |")
        builder.appendLine("|--------|-------|")
        builder.appendLine("| Total Tests | $totalTests |")
        builder.appendLine("| Passed | $passedTests |")
        builder.appendLine("| Failed | $failedTests |")
        builder.appendLine()
        
        // By Section
        builder.appendLine("## Results by Section")
        builder.appendLine()
        
        results.values
            .groupBy { it.section }
            .forEach { (section, sectionResults) ->
                val sectionTotal = sectionResults.size
                val sectionPassed = sectionResults.count { it.passed }
                val sectionPercentage = (sectionPassed.toDouble() / sectionTotal * 100).toInt()
                
                builder.appendLine("### Section $section: **$sectionPercentage%**")
                builder.appendLine()
                builder.appendLine("| Subsection | Test | Result | Errors |")
                builder.appendLine("|------------|------|--------|--------|")
                
                sectionResults
                    .sortedBy { "${it.subsection}.${it.testName}" }
                    .forEach { result ->
                        val status = if (result.passed) "✅ PASS" else "❌ FAIL"
                        val errorsDisplay = if (result.errors.isNotEmpty()) {
                            result.errors.joinToString(", ")
                        } else {
                            "-"
                        }
                        
                        builder.appendLine("| ${result.subsection} | ${result.testName} | $status | $errorsDisplay |")
                    }
                
                builder.appendLine()
            }
        
        // Failed Tests Details
        if (failedTests > 0) {
            builder.appendLine("## Failed Tests Details")
            builder.appendLine()
            
            results.values
                .filter { !it.passed }
                .forEach { result ->
                    builder.appendLine("### ${result.section}.${result.subsection}.${result.testName}")
                    builder.appendLine()
                    builder.appendLine("**WGSL Source:**")
                    builder.appendLine("```wgsl")
                    builder.appendLine(result.wgslSource ?: "N/A")
                    builder.appendLine("```")
                    builder.appendLine()
                    builder.appendLine("**Errors:**")
                    result.errors.forEach { error ->
                        builder.appendLine("- $error")
                    }
                    builder.appendLine()
                }
        }
        
        // Conformance Summary
        builder.appendLine("## Conformance Summary")
        builder.appendLine()
        
        if (conformancePercentage == 100) {
            builder.appendLine("✅ **The parser is 100% compliant with the WGSL CRD specification!**")
        } else {
            builder.appendLine("⚠️ **The parser is $conformancePercentage% compliant with the WGSL CRD specification.**")
            builder.appendLine()
            builder.appendLine("To achieve 100% conformance, the following areas need attention:")
            builder.appendLine()
            
            results.values
                .filter { !it.passed }
                .groupBy { it.section }
                .forEach { (section, failures) ->
                    builder.appendLine("- **Section $section**: ${failures.size} failing tests")
                }
        }
        
        return builder.toString()
    }
    
    /**
     * Clears all recorded results (useful for testing).
     */
    fun clearResults() {
        results.clear()
    }
    
    /**
     * Gets the current version of the parser.
     */
    private fun getVersion(): String {
        // TODO: Implement version detection from build properties
        return "2.0.0-SNAPSHOT"
    }
    
    /**
     * Exports the report to a file.
     */
    fun exportReport(filePath: String) {
        val report = generateReport()
        java.io.File(filePath).writeText(report)
    }
}

package io.ygdrasil.wgsl.parser.conformance

import io.kotest.core.spec.style.FunSpec
import io.ygdrasil.wgsl.parser.conformance.ConformanceReporter.testConformance

/**
 * Conformance tests for WGSL Section 3: Lexical Conventions
 */
class LexicalConformanceTest : FunSpec({
    
    // Section 3.1: Tokens
    context("Section 3.1: Tokens") {
        test("3.1.1 - Identifiers") {
            val passed = testConformance(
                section = "3",
                subsection = "3.1",

                source = """
                    let myVariable = 42;
                    let _private = 10;
                    let $special = 20;
                    fn myFunction() {}

            passed shouldBe true
        }

            val passed = testConformance(
                section = "3",
                subsection = "3.1",

### 10. Script de vérification de conformité

```bash
#!/bin/bash
# check-conformance.sh

# Ce script vérifie que la conformité est à 100%
# Il est appelé dans le workflow GitHub Actions

cd wgsl/parser

# Exécuter les tests de conformité
./gradlew test --tests "io.ygdrasil.wgsl.parser.conformance.*" > /tmp/conformance-output.txt 2>&1

# Extraire le pourcentage de conformité
echo "Checking conformance percentage..."

# Chercher la ligne avec "Overall Conformance"
CONFORMANCE_LINE=$(grep "Overall Conformance" /tmp/conformance-output.txt)

if [ -z "$CONFORMANCE_LINE" ]; then
    echo "ERROR: Could not find conformance percentage in output"
    cat /tmp/conformance-output.txt
    exit 1
fi

echo "Conformance line: $CONFORMANCE_LINE"

# Extraire le pourcentage
PERCENTAGE=$(echo "$CONFORMANCE_LINE" | grep -oP '\d+%' | tr -d '%')

if [ -z "$PERCENTAGE" ]; then
    echo "ERROR: Could not extract percentage from: $CONFORMANCE_LINE"
    exit 1
fi

echo "Detected conformance: $PERCENTAGE%"

# Vérifier que c'est 100%
if [ "$PERCENTAGE" -eq 100 ]; then
    echo "✅ Conformance is at 100% - PASS"
    exit 0
else
    echo "❌ Conformance is at ${PERCENTAGE}% - FAIL (expected 100%)"
    exit 1
fi
