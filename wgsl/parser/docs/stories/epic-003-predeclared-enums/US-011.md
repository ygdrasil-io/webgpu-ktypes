```kotlin
package io.ygdrasil.wgsl.ast

import io.ygdrasil.wgsl.ir.Span

/**
 * Base class for all predeclared enumerants.
 * 
 * Predeclared enumerants are built-in enum-like values defined by the WGSL
 * specification. They are used in various WGSL constructs like texture
 * sampling, interpolation, etc.
 * 
 * @property category The category/group this enumerant belongs to (e.g., "AddressMode")
 * @property value The string value of this enumerant (e.g., "clamp_to_edge")
 * @property displayName Human-readable display name
 */
sealed class PredeclaredEnumerant {
    abstract val category: String
    abstract val value: String
    abstract val displayName: String
    
    /**
     * Returns the fully qualified name of this enumerant.
     * Format: "Category.Value"
     */
    fun getQualifiedName(): String = "$category.$value"
    
    /**
     * Checks if this enumerant belongs to the specified category.
     */
    fun isInCategory(category: String): Boolean = this.category == category
}

// =============================================================================
// TEXTURE SAMPLING ENUMERANTS
// =============================================================================

/**
 * Address mode for texture coordinate wrapping.
 * 
 * Defines how texture coordinates outside the [0, 1] range are handled.
 * 
 * WGSL Spec: https://www.w3.org/TR/WGSL/#address-mode
 */
sealed class AddressMode(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "AddressMode"
    override val displayName: String = value
    
    object CLAMP_TO_EDGE : AddressMode("clamp_to_edge")
    object REPEAT : AddressMode("repeat")
    object MIRROR_REPEAT : AddressMode("mirror_repeat")
    
    companion object {
        /** All valid AddressMode values. */
        val ALL: List<AddressMode> = listOf(CLAMP_TO_EDGE, REPEAT, MIRROR_REPEAT)
        
        /** Gets an AddressMode by its string value. */
        fun fromValue(value: String): AddressMode? = ALL.find { it.value == value }
    }
}

/**
 * Filter mode for texture sampling.
 * 
 * Defines the filtering algorithm used when sampling textures.
 * 
 * WGSL Spec: https://www.w3.org/TR/WGSL/#filter-mode
 */
sealed class FilterMode(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "FilterMode"
    override val displayName: String = value
    
    object NEAREST : FilterMode("nearest")
    object LINEAR : FilterMode("linear")
    
    companion object {
        val ALL: List<FilterMode> = listOf(NEAREST, LINEAR)
        fun fromValue(value: String): FilterMode? = ALL.find { it.value == value }
    }
}

/**
 * Mipmap filter mode.
 * 
 * Defines the filtering algorithm used between mipmap levels.
 * 
 * WGSL Spec: https://www.w3.org/TR/WGSL/#mipmap-filter-mode
 */
sealed class MipmapFilterMode(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "MipmapFilterMode"
    override val displayName: String = value
    
    object NEAREST : MipmapFilterMode("nearest")
    object LINEAR : MipmapFilterMode("linear")
    
    companion object {
        val ALL: List<MipmapFilterMode> = listOf(NEAREST, LINEAR)
        fun fromValue(value: String): MipmapFilterMode? = ALL.find { it.value == value }
    }
}

// =============================================================================
// INTERPOLATION ENUMERANTS
// =============================================================================

/**
 * Interpolation type.
 * 
 * Defines how values are interpolated across a triangle or line.
 * 
 * WGSL Spec: https://www.w3.org/TR/WGSL/#interpolation-type
 */
sealed class InterpolationType(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "InterpolationType"
    override val displayName: String = value
    
    object PERSPECTIVE : InterpolationType("perspective")
    object LINEAR : InterpolationType("linear")
    object FLAT : InterpolationType("flat")
    
    companion object {
        val ALL: List<InterpolationType> = listOf(PERSPECTIVE, LINEAR, FLAT)
        fun fromValue(value: String): InterpolationType? = ALL.find { it.value == value }
    }
}

/**
 * Interpolation sampling.
 * 
 * Defines the sample location used for interpolation.
 * 
 * WGSL Spec: https://www.w3.org/TR/WGSL/#interpolation-sampling
 */
sealed class InterpolationSampling(override val value: String) : PredeclaredEnumerant() {
    override val category: String = "InterpolationSampling"
    override val displayName: String = value
    
    object CENTER : InterpolationSampling("center")
    object CENTROID : InterpolationSampling("centroid")
    object SAMPLE : InterpolationSampling("sample")
    
    companion object {
        val ALL: List<InterpolationSampling> = listOf(CENTER, CENTROID, SAMPLE)
        fun fromValue(value: String): InterpolationSampling? = ALL.find { it.value == value }
    }
}

// =============================================================================
// SWIZZLE NAMES (already in TokenKind)
// =============================================================================

/**
 * Predefined swizzle component names.
 * 
 * These are used for vector component access (e.g., v.x, v.y, v.r, etc.)
 */
object SwizzleNames {
    val X: String = "x"
    val Y: String = "y"
    val Z: String = "z"
    val W: String = "w"
    
    // Aliases
    val R: String = "r"
    val G: String = "g"
    val B: String = "b"
    val A: String = "a"
    
    // Texture coordinates
    val S: String = "s"
    val T: String = "t"
    val P: String = "p"
    val Q: String = "q"
    
    val ALL: Set<String> = setOf(X, Y, Z, W, R, G, B, A, S, T, P, Q)
    
    fun isValidSwizzle(name: String): Boolean = ALL.contains(name)
}

// =============================================================================
// REGISTRY (pour accès global)
// =============================================================================

/**
 * Registry of all predeclared enumerants.
 * 
 * Provides a centralized way to access all predeclared enumerants and
 * perform lookups by qualified name (e.g., "AddressMode.clamp_to_edge").
 */
object PredeclaredEnumerantRegistry {
    
    private val enumerantsByCategory: Map<String, List<PredeclaredEnumerant>> = mapOf(
        "AddressMode" to AddressMode.ALL,
        "FilterMode" to FilterMode.ALL,
        "MipmapFilterMode" to MipmapFilterMode.ALL,
        "InterpolationType" to InterpolationType.ALL,
        "InterpolationSampling" to InterpolationSampling.ALL,
    )
    
    private val enumerantsByQualifiedName: Map<String, PredeclaredEnumerant> by lazy {
        enumerantsByCategory.flatMap { (category, enumerants) ->
            enumerants.map { enumerant ->
                "$category.${enumerant.value}" to enumerant
            }
        }.toMap()
    }
    
    /**
     * Gets all categories of predeclared enumerants.
     */
    fun getCategories(): List<String> = enumerantsByCategory.keys.toList()
    
    /**
     * Gets all enumerants in a specific category.
     */
    fun getEnumerantsInCategory(category: String): List<PredeclaredEnumerant>? {
        return enumerantsByCategory[category]
    }
    
    /**
     * Gets all enumerants across all categories.
     */
    fun getAllEnumerants(): List<PredeclaredEnumerant> {
        return enumerantsByCategory.values.flatten()
    }
    
    /**
     * Gets a predeclared enumerant by its qualified name.
     * 
     * @param qualifiedName Format: "Category.Value" (e.g., "AddressMode.clamp_to_edge")
     * @return The enumerant, or null if not found
     */
    fun getEnumerantByQualifiedName(qualifiedName: String): PredeclaredEnumerant? {
        return enumerantsByQualifiedName[qualifiedName]
    }
    
    /**
     * Checks if a qualified name refers to a valid predeclared enumerant.
     */
    fun isValidQualifiedName(qualifiedName: String): Boolean {
        return enumerantsByQualifiedName.containsKey(qualifiedName)
    }
    
    /**
     * Gets a predeclared enumerant by category and value.
     */
    fun getEnumerant(category: String, value: String): PredeclaredEnumerant? {
        return enumerantsByCategory[category]?.find { it.value == value }
    }
}

// =============================================================================
// EXTENSION FUNCTIONS
// =============================================================================

/**
 * Checks if this string is a valid predefined swizzle name.
 */
fun String.isValidSwizzleName(): Boolean = SwizzleNames.isValidSwizzle(this)

/**
 * Parses a qualified enumerant name and returns the enumerant.
 * 
 * @param qualifiedName Format: "Category.Value"
 * @return The enumerant, or null if not found or invalid format
 */
fun String.toPredeclaredEnumerant(): PredeclaredEnumerant? {
    val parts = this.split('.', limit = 2)
    if (parts.size != 2) return null
    
    return PredeclaredEnumerantRegistry.getEnumerant(parts[0], parts[1])
}
```

```kotlin
// PredeclaredEnumsTest.kt
package io.ygdrasil.wgsl.ast

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldContain
import io.kotest.matchers.shouldHaveSize
import io.kotest.matchers.shouldNotBe

/**
 * Tests for predeclared enumerants.
 */
class PredeclaredEnumsTest : FunSpec({
    
    context("AddressMode") {
        test("CLAMP_TO_EDGE properties") {
            AddressMode.CLAMP_TO_EDGE.category shouldBe "AddressMode"
            AddressMode.CLAMP_TO_EDGE.value shouldBe "clamp_to_edge"
            AddressMode.CLAMP_TO_EDGE.displayName shouldBe "clamp_to_edge"
        }
        
        test("REPEAT properties") {
            AddressMode.REPEAT.value shouldBe "repeat"
        }
        
        test("MIRROR_REPEAT properties") {
            AddressMode.MIRROR_REPEAT.value shouldBe "mirror_repeat"
        }
        
        test("ALL contains all values") {
            AddressMode.ALL shouldHaveSize 3
            AddressMode.ALL shouldContain AddressMode.CLAMP_TO_EDGE
            AddressMode.ALL shouldContain AddressMode.REPEAT
            AddressMode.ALL shouldContain AddressMode.MIRROR_REPEAT
        }
        
        test("fromValue works") {
            AddressMode.fromValue("clamp_to_edge") shouldBe AddressMode.CLAMP_TO_EDGE
            AddressMode.fromValue("repeat") shouldBe AddressMode.REPEAT
            AddressMode.fromValue("invalid") shouldBe null
        }
    }
    
    context("FilterMode") {
        test("NEAREST and LINEAR") {
            FilterMode.NEAREST.value shouldBe "nearest"
            FilterMode.LINEAR.value shouldBe "linear"
        }
        
        test("fromValue works") {
            FilterMode.fromValue("nearest") shouldBe FilterMode.NEAREST
            FilterMode.fromValue("linear") shouldBe FilterMode.LINEAR
        }
    }
    
    context("PredeclaredEnumerant base class") {
        test("getQualifiedName") {
            AddressMode.CLAMP_TO_EDGE.getQualifiedName() shouldBe "AddressMode.clamp_to_edge"
            FilterMode.NEAREST.getQualifiedName() shouldBe "FilterMode.nearest"
        }
        
        test("isInCategory") {
            AddressMode.CLAMP_TO_EDGE.isInCategory("AddressMode") shouldBe true
            AddressMode.CLAMP_TO_EDGE.isInCategory("FilterMode") shouldBe false
        }
    }
    
    context("PredeclaredEnumerantRegistry") {
        test("getCategories") {
            val categories = PredeclaredEnumerantRegistry.getCategories()
            categories shouldContain "AddressMode"
            categories shouldContain "FilterMode"
            categories shouldContain "MipmapFilterMode"
            categories shouldContain "InterpolationType"
            categories shouldContain "InterpolationSampling"
        }
        
        test("getEnumerantsInCategory") {
            val addressModes = PredeclaredEnumerantRegistry.getEnumerantsInCategory("AddressMode")
            addressModes shouldNotBe null
            addressModes shouldHaveSize 3
        }
        
        test("getAllEnumerants") {
            val all = PredeclaredEnumerantRegistry.getAllEnumerants()
            all.isNotEmpty() shouldBe true
        }
        
        test("getEnumerantByQualifiedName") {
            val enumerant = PredeclaredEnumerantRegistry.getEnumerantByQualifiedName("AddressMode.clamp_to_edge")
            enumerant shouldBe AddressMode.CLAMP_TO_EDGE
        }
        
        test("isValidQualifiedName") {
            PredeclaredEnumerantRegistry.isValidQualifiedName("AddressMode.clamp_to_edge") shouldBe true
            PredeclaredEnumerantRegistry.isValidQualifiedName("Invalid.Category") shouldBe false
        }
        
        test("getEnumerant") {
            val enumerant = PredeclaredEnumerantRegistry.getEnumerant("AddressMode", "clamp_to_edge")
            enumerant shouldBe AddressMode.CLAMP_TO_EDGE
        }
    }
    
    context("Extension functions") {
        test("isValidSwizzleName") {
            "x".isValidSwizzleName() shouldBe true
            "y".isValidSwizzleName() shouldBe true
            "z".isValidSwizzleName() shouldBe true
            "w".isValidSwizzleName() shouldBe true
            "r".isValidSwizzleName() shouldBe true
            "g".isValidSwizzleName() shouldBe true
            "b".isValidSwizzleName() shouldBe true
            "a".isValidSwizzleName() shouldBe true
            "invalid".isValidSwizzleName() shouldBe false
        }
        
        test("toPredeclaredEnumerant") {
            "AddressMode.clamp_to_edge".toPredeclaredEnumerant() shouldBe AddressMode.CLAMP_TO_EDGE
            "FilterMode.nearest".toPredeclaredEnumerant() shouldBe FilterMode.NEAREST
            "Invalid.Category".toPredeclaredEnumerant() shouldBe null
            "AddressMode".toPredeclaredEnumerant() shouldBe null // Missing value
        }
    }
})
```
