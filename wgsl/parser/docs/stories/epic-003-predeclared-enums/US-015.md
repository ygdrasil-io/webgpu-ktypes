package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.collections.shouldContain
import io.kotest.matchers.collections.shouldHaveSize

import io.kotest.matchers.shouldNotBe
import io.ygdrasil.wgsl.ast.*

/**
 * Tests for the PredeclaredEnumerantRegistry.
 * Verifies that all predefined enumerants are correctly registered.
 */
class PredeclaredEnumerantRegistryTest : FunSpec({
    
    context("registry initialization") {
        test("all expected categories are registered") {
            val expectedCategories = listOf(
                "AddressMode",
                "FilterMode", 
                "MipmapFilterMode",
                "InterpolationType",
                "InterpolationSampling"
            )
            
            expectedCategories.forEach { category ->
                PredeclaredEnumerantRegistry.isKnownCategory(category) shouldBe true
            }
        }
        
        test("unknown category returns false") {
            PredeclaredEnumerantRegistry.isKnownCategory("UnknownCategory") shouldBe false
        }
    }
    
    context("AddressMode") {
        test("has all required values") {
            val values = listOf("clamp_to_edge", "repeat", "mirror_repeat")
            val addressModes = PredeclaredEnumerantRegistry.getEnumerantsByCategory("AddressMode")
            
            addressModes shouldHaveSize 3
            values.forEach { value ->
                PredeclaredEnumerantRegistry.getEnumerant("AddressMode", value) shouldNotBe null
            }
        }
        
        test("getValidValues returns all values") {
            val values = PredeclaredEnumerantRegistry.getValidValues("AddressMode")
            values shouldHaveSize 3
            values shouldContain "clamp_to_edge"
            values shouldContain "repeat"
            values shouldContain "mirror_repeat"
        }
        
        test("getEnumerant returns correct enumerant") {
            val enumerant = PredeclaredEnumerantRegistry.getEnumerant("AddressMode", "clamp_to_edge")
            enumerant shouldNotBe null
            enumerant?.name shouldBe "clamp_to_edge"
            enumerant?.category shouldBe "AddressMode"
            enumerant?.getQualifiedName() shouldBe "AddressMode.clamp_to_edge"
        }
        
        test("getEnumerant returns null for invalid value") {
            PredeclaredEnumerantRegistry.getEnumerant("AddressMode", "invalid") shouldBe null
        }
    }

            val values = listOf("nearest", "linear")

            values.forEach { value ->
                PredeclaredEnumerantRegistry.getEnumerant("InterpolationType", value) shouldNotBe null
            }
        }
    }
    
    context("InterpolationSampling") {
        test("has all required values") {
            val values = listOf("sample", "vertex", "centroid")
            val samplingModes = PredeclaredEnumerantRegistry.getEnumerantsByCategory("InterpolationSampling")

package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.lexer.Lexer

/**
 * Integration tests for predeclared enumerants with other parser features.
 */
class PredeclaredEnumerantIntegrationTest : FunSpec({
    
    context("predeclared enumerants with user-defined enums") {
        test("both can coexist") {
            val source = """
                enum MyEnum { VALUE1, VALUE2 }
                let userEnum = MyEnum.VALUE1;
                let preEnum = AddressMode.clamp_to_edge;
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("similar syntax works for both") {
            val source = """
                enum Color { RED, GREEN }
                let c1 = Color.RED;
                let c2 = AddressMode.REPEAT;  // Note: REPEAT n'existe pas, mais AddressMode.repeat oui
            """.trimIndent()
            
            // Cela devrait avoir une erreur pour AddressMode.REPEAT
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            // REPEAT n'est pas valide (c'est 'repeat' en minuscule)
            parser.errors.isNotEmpty() shouldBe true
        }
    }
    
    context("predeclared enumerants with type aliases") {
        test("type alias with AddressMode") {
            val source = """
                alias Mode = AddressMode;
                let mode: Mode = AddressMode.clamp_to_edge;
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("predeclared enumerants in generic contexts") {
        test("enumerant as template argument") {
            // Note: WGSL ne supporte pas les templates génériques comme C++
            // Mais les énumérants peuvent être utilisés dans des contextes génériques
            val source = """
                fn generic<T>(value: T) {}
                fn main() {
                    generic<AddressMode>(AddressMode.clamp_to_edge);
                }
            """.trimIndent()
            // Cela dépend de la support des templates dans le parser
            // Peut générer une erreur si les templates ne sont pas supportés
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            // On vérifie juste que ça parse
        }
    }
    
    context("predeclared enumerants with attributes") {
        test("enumerant with stage attribute") {
            val source = """
                [[stage(vertex)]]
                fn main() {
                    let mode = AddressMode.clamp_to_edge;
                }
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("predeclared enumerants in compute shaders") {
        test("enumerant in compute shader") {
            val source = """
                [[stage(compute), workgroup_size(1)]]
                fn main() {
                    let mode = AddressMode.clamp_to_edge;
                }
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("real-world usage scenarios") {
        test("texture sampling with enumerants") {
            val source = """
                [[group(0), binding(0)]]
                var texture: texture_2d<f32>;
                
                [[group(0), binding(1)]]
                var sampler: sampler;
                
                [[stage(fragment)]]
                fn main() {
                    let sampled = textureSample(texture, sampler, textureCoordinate().xy, AddressMode.clamp_to_edge, FilterMode.linear);
                }
            """.trimIndent()
            // Note: textureSample est une fonction intrinsèque
            // Cela peut générer des erreurs si les fonctions intrinsèques ne sont pas supportées
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            // On vérifie que les énumérants parsent correctement
        }
    }
    
    context("edge cases") {
        test("nested member access with enumerant") {
            // Tester qu'on ne confond pas les énumérants avec d'autres member accesses
            val source = """
                struct Outer {
                    inner: Inner,
                }
                struct Inner {
                    mode: AddressMode,
                }
                let o: Outer;
                let mode = o.inner.mode;
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("enumerant in complex expression") {
            val source = """
                let mode = AddressMode.clamp_to_edge;
                let result = (mode == AddressMode.clamp_to_edge) && (mode != AddressMode.repeat);
            """.trimIndent()
