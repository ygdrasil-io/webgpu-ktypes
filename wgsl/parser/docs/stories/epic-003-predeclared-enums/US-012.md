// Ajouter une nouvelle classe pour représenter les références aux énumérants prédéclarés

/**
 * Represents a reference to a predeclared enumerant value.
 * 
 * For example: AddressMode.clamp_to_edge
 * 
 * @property baseType The type that contains the member (e.g., AddressMode)
 * @property enumerant The actual predeclared enumerant value
 * @property span The source span of this expression
 */
data class PredeclaredEnumerantExpr(
    val category: String,
    val enumerant: PredeclaredEnumerant,
    override val span: Span,
) : Expression() {
    /**
     * Returns the fully qualified name of this enumerant.
     */
    fun getQualifiedName(): String = enumerant.getQualifiedName()
    
    override fun <R> accept(visitor: ExpressionVisitor<R>): R {
        return visitor.visitPredeclaredEnumerant(this)
    }
}

// Étendre ExpressionVisitor pour supporter les nouveaux types d'expressions
interface ExpressionVisitor<R> {
    // ... méthodes existantes
    
    /**
     * Visits a predeclared enumerant expression.
     */
    fun visitPredeclaredEnumerant(expr: PredeclaredEnumerantExpr): R
    
    /**
     * Visits an enum member expression (pour les enum définis par l'utilisateur).
     */
    fun visitEnumMember(expr: EnumMemberExpr): R
}

// Mettre à jour DefaultExpressionVisitor
class DefaultExpressionVisitor<R> : ExpressionVisitor<R> {
    // ... implémentations existantes
    
    override fun visitPredeclaredEnumerant(expr: PredeclaredEnumerantExpr): R = defaultResult()
    override fun visitEnumMember(expr: EnumMemberExpr): R = defaultResult()
}
```

### 2. Modifications dans Parser.kt

// Modifier parseMemberAccess() pour gérer les énumérants prédéclarés

/**
 * Parses a member access expression (e.g., x.y, AddressMode.clamp_to_edge).
 */
private fun parseMemberAccess(base: Expression): Expression {
    val start = base.span
    
    expect(TokenKind.DOT)
    val memberName = expectIdentifier()
    
    // Vérifier si la base est une référence à une catégorie d'enum prédéfini
    if (base is IdentExpr) {
        val category = base.identifier
        
        // Vérifier si c'est un énum prédéfini
        val enumerant = PredeclaredEnumerantRegistry.getEnumerant(category, memberName)
        if (enumerant != null) {
            return PredeclaredEnumerantExpr(category, enumerant, spanFrom(start))
        }
        
        // Vérifier si c'est un enum défini par l'utilisateur
        // (Cela nécessitera une résolution de type plus tard)
    }
    
    // Sinon, créer un MemberAccessExpr normal
    return MemberAccessExpr(base, memberName, spanFrom(start))
}

// Ajouter une méthode pour valider les références aux énumérants
/**
 * Validates that a predeclared enumerant reference is valid.
 * Called during semantic analysis.
 */
fun validatePredeclaredEnumerant(expr: PredeclaredEnumerantExpr): Boolean {
    return PredeclaredEnumerantRegistry.isValidQualifiedName(expr.getQualifiedName())
}
```
