// ErrorMessages.kt
object ErrorMessages {
    
    // Erreurs pour les énumérants prédéclarés
    fun invalidPredeclaredEnumerantValue(
        category: String,
        invalidValue: String,
        validValues: List<String>
    ): String {
        return "Invalid ${category} value '$invalidValue'. " +
               "Valid values are: ${validValues.joinToString(", ")}. " +
               "See WGSL specification for more details."
    }
    
    fun unknownPredeclaredEnumerantCategory(category: String): String {
        return "Unknown predeclared enumerant category '$category'. " +
               "Category not found in the WGSL specification."
    }
    
    fun enumerantTypeMismatch(expected: String, actual: String): String {
        return "Type mismatch: expected '$expected' but found '$actual'"
    }
}

// SemanticValidator.kt
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.parser.error

/**
 * Validates semantic correctness of the parsed AST.
 * This includes validation of predeclared enumerant values.
 */
class SemanticValidator(private val translationUnit: TranslationUnit) {
    
    private val errors = mutableListOf<String>()
    
    fun validate(): List<String> {
        translationUnit.accept(this)
        return errors
    }
    
    fun hasErrors(): Boolean = errors.isNotEmpty()
    
    private fun error(message: String, span: Span) {
        errors.add("[${span.start.line}:${span.start.column}] $message")
    }
    
    // Visitor pattern implementation
    fun visitTranslationUnit(unit: TranslationUnit) {
        unit.declarations.forEach { declaration ->
            when (declaration) {
                is GlobalVariableDecl -> visitGlobalVariable(declaration)
                is FunctionDecl -> visitFunction(declaration)
                is StructDecl -> visitStruct(declaration)
                // ... autres types de déclarations
            }
        }
    }
    
    private fun visitExpression(expr: Expression) {
        when (expr) {
            is PredeclaredEnumerantExpr -> validatePredeclaredEnumerant(expr)
            is MemberAccessExpr -> {
                // Valider les member access qui pourraient être des énumérants
                if (expr.base is IdentExpr) {
                    validateMemberAccessAsEnumerant(expr)
                }
            }
            // Visiter les expressions enfants
            else -> {
                expr.children().forEach { child ->
                    if (child is Expression) {
                        visitExpression(child)
                    }
                }
            }
        }
    }
    
    /**
     * Valide qu'une référence à un énumérant prédéclaré est valide.
     */
    private fun validatePredeclaredEnumerant(expr: PredeclaredEnumerantExpr) {
        if (!PredeclaredEnumerantRegistry.isValidQualifiedName(expr.getQualifiedName())) {
            error(
                ErrorMessages.invalidPredeclaredEnumerantValue(
                    expr.category,
                    expr.enumerant.value,
                    PredeclaredEnumerantRegistry.getValidValues(expr.category)
                ),
                expr.span
            )
        }
    }
    
    /**
     * Valide qu'un member access fait référence à un énumérant valide.
     */
    private fun validateMemberAccessAsEnumerant(expr: MemberAccessExpr) {
        val base = expr.base
        if (base !is IdentExpr) return
        
        val category = base.identifier
        val member = expr.member
        
        // Vérifier si c'est une catégorie d'enum connue
        if (PredeclaredEnumerantRegistry.isKnownCategory(category)) {
            val enumerant = PredeclaredEnumerantRegistry.getEnumerant(category, member)
            if (enumerant == null) {
                error(
                    ErrorMessages.invalidPredeclaredEnumerantValue(
                        category,
                        member,
                        PredeclaredEnumerantRegistry.getValidValues(category)
                    ),
                    expr.span
                )
            }
        }
    }
    
    private fun visitGlobalVariable(decl: GlobalVariableDecl) {
        visitExpression(decl.expr)
    }
    
    private fun visitFunction(decl: FunctionDecl) {
        // Valider les paramètres
        decl.parameters.forEach { param ->
            if (param.expr != null) {
                visitExpression(param.expr)
            }
        }
        // Valider le corps de la fonction
        decl.body?.statements?.forEach { stmt ->
            visitStatement(stmt)
        }
    }
    
    private fun visitStatement(stmt: Statement) {
        when (stmt) {
            is ExpressionStmt -> visitExpression(stmt.expr)
            is AssignStmt -> {
                visitExpression(stmt.lhs)
                visitExpression(stmt.rhs)
            }
            is ReturnStmt -> {
                stmt.expr?.let { visitExpression(it) }
            }
            is IfStmt -> {
                visitExpression(stmt.condition)
                stmt.body?.statements?.forEach { visitStatement(it) }
                stmt.elseBody?.statements?.forEach { visitStatement(it) }
            }
            // ... autres types de statements
        }
    }
    
    private fun visitStruct(decl: StructDecl) {
        decl.members.forEach { member ->
            if (member.expr != null) {
                visitExpression(member.expr)
            }
        }
    }
}

### 3. Intégration dans Parser.kt

// Parser.kt - Ajouter la validation sémantique

class Parser(private val lexer: Lexer) {
    // ... code existant
    
    /**
     * Parse the input and validate semantic correctness.
     * This is the main entry point for parsing with validation.
     */
    fun parseAndValidate(): TranslationUnit {
        val unit = parse()
        
        // Valider sémantiquement
        val validator = SemanticValidator(unit)
        val semanticErrors = validator.validate()
        
        // Ajouter les erreurs sémantiques aux erreurs de parsing
        errors.addAll(semanticErrors)
        
        return unit
    }
    
    /**
     * Parse without semantic validation (faster, for when you know it's valid).
     */
    fun parse(): TranslationUnit {
        // ... implémentation existante
    }
    
    /**
     * Valide une PredeclaredEnumerantExpr directement pendant le parsing.
     */
    private fun validatePredeclaredEnumerant(expr: PredeclaredEnumerantExpr) {
        val validator = SemanticValidator(translationUnit)
        validator.validatePredeclaredEnumerant(expr)
        errors.addAll(validator.errors)
    }
}

// PredeclaredEnumerantRegistry.kt - Ajouter des méthodes utilitaires

object PredeclaredEnumerantRegistry {
    // ... code existant
    
    /**
     * Returns true if the category is known.
     */
    fun isKnownCategory(category: String): Boolean {
        return categories.contains(category)
    }
    
    /**
     * Returns all valid values for a given category.
     */
    fun getValidValues(category: String): List<String> {
        return enumerants
            .filter { it.category == category }
            .map { it.value }
    }
    
    /**
     * Returns true if the qualified name is valid.
     */
    fun isValidQualifiedName(qualifiedName: String): Boolean {
        return enumerants.any { it.getQualifiedName() == qualifiedName }
    }
    
    /**
     * Returns all categories defined in the registry.
     */
    val categories: Set<String>
        get() = enumerants.map { it.category }.toSet()
}

// PredeclaredEnumerantValidationTest.kt
package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.collections.shouldContain
import io.kotest.matchers.collections.shouldHaveSize
import io.kotest.matchers.shouldBe
import io.kotest.matchers.string.shouldContain
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.lexer.Lexer

/**
 * Tests for validation of predeclared enumerant values.
 */
class PredeclaredEnumerantValidationTest : FunSpec({
    
    context("valid enumerant values") {
        test("AddressMode.clamp_to_edge is valid") {
            val source = "let mode = AddressMode.clamp_to_edge;"
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("FilterMode.nearest is valid") {
            val source = "let filter = FilterMode.nearest;"
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("all AddressMode values are valid") {
            val values = listOf("clamp_to_edge", "repeat", "mirror_repeat")
            values.forEach { value ->
                val source = "let mode = AddressMode.$value;"
                val parser = Parser(Lexer(source))
                val unit = parser.parseAndValidate()
                parser.errors.isEmpty() shouldBe true
            }
        }
        
        test("all FilterMode values are valid") {
            val values = listOf("nearest", "linear")
            values.forEach { value ->
                val source = "let filter = FilterMode.$value;"
                val parser = Parser(Lexer(source))
                val unit = parser.parseAndValidate()
                parser.errors.isEmpty() shouldBe true
            }
        }
    }
    
    context("invalid enumerant values") {
        test("AddressMode.invalid_value generates error") {
            val source = "let mode = AddressMode.invalid_value;"
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isNotEmpty() shouldBe true
            parser.errors shouldContain "invalid_value"
            parser.errors shouldContain "AddressMode"
        }
        
        test("FilterMode.invalid generates error") {
            val source = "let filter = FilterMode.invalid;"
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isNotEmpty() shouldBe true
            parser.errors shouldContain "invalid"
            parser.errors shouldContain "FilterMode"
        }
        
        test("unknown category generates error") {
            val source = "let x = UnknownCategory.value;"
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            // Cela sera parsé comme un MemberAccessExpr, mais devra être validé
            parser.errors.isNotEmpty() shouldBe true
        }
        
        test("error message includes valid values") {
            val source = "let mode = AddressMode.invalid;"
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isNotEmpty() shouldBe true
            parser.errors.first() shouldContain "clamp_to_edge"
            parser.errors.first() shouldContain "repeat"
            parser.errors.first() shouldContain "mirror_repeat"
        }
    }
    
    context("enumerants in function parameters") {
        test("valid enumerant in function parameter") {
            val source = """
                fn sample(mode: AddressMode = AddressMode.clamp_to_edge) {
                }
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("invalid enumerant in function parameter") {
            val source = """
                fn sample(mode: AddressMode = AddressMode.invalid) {
                }
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isNotEmpty() shouldBe true
        }
    }
    
    context("enumerants in struct initialization") {
        test("valid enumerant in struct initializer") {
            val source = """
                struct Config {
                    mode: AddressMode,
                }
                let config = Config(AddressMode.repeat);
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("invalid enumerant in struct initializer") {
            val source = """
                struct Config {
                    mode: AddressMode,
                }
                let config = Config(AddressMode.invalid_value);
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isNotEmpty() shouldBe true
        }
    }
    
    context("multiple errors") {
        test("multiple invalid enumerant values") {
            val source = """
                let mode1 = AddressMode.invalid1;
                let mode2 = AddressMode.invalid2;
            """.trimIndent()
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors shouldHaveSize 2
        }
    }
})
