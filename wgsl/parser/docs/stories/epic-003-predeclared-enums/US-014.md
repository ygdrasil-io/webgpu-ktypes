## 📋 Description

**En tant que** : Développeur utilisant à la fois les enum et les prédeclared enumerants  
**Je veux** : Une expérience cohérente entre les deux systèmes  
**Afin de** : Avoir un système de types unifié et éviter la duplication de code

---

## ✅ Critères d'Acceptation

- [ ] Les enum définis par l'utilisateur fonctionnent comme les prédeclared enumerants
- [ ] Même syntaxe pour accéder aux valeurs (`MyEnum.VALUE` et `AddressMode.clamp_to_edge`)
- [ ] Même sémantique de type
- [ ] Les deux systèmes partagent la même infrastructure de base
- [ ] Test: Utilisation cohérente des deux types d'enum dans le même code
- [ ] Test: Pas de différence de comportement entre les enum user-defined et prédéclarés

```kotlin
// Enum.kt
package io.ygdrasil.wgsl.ast

/**
 * Represents a single value within an enum (either user-defined or predeclared).
 * This is the common base for all enum values.
 */
sealed interface EnumValue {
    val name: String
    val value: Expression?
    val attributes: List<Attribute>
    val span: Span
    
    /**
     * Returns the fully qualified name of this enum value.
     */
    fun getQualifiedName(): String
}

/**
 * Represents a member of a user-defined enum.
 */
data class EnumMember(
    override val name: String,
    override val value: Expression?,  // Can be null for auto-assigned values
    override val attributes: List<Attribute>,
    override val span: Span,
) : EnumValue {
    override fun getQualifiedName(): String = name
}

/**
 * Represents a predefined/predeclared enumerant value.
 * These are defined by the WGSL specification and not by user code.
 */
data class PredeclaredEnumerant(
    val category: String,
    override val name: String,
    override val value: Expression?,  // Typically null - values are symbolic
    override val attributes: List<Attribute> = emptyList(),
    override val span: Span = Span.INVALID,
) : EnumValue {
    override fun getQualifiedName(): String = "$category.$name"
    
    companion object {
        /**
         * Creates a predeclared enumerant with a symbolic value.
         */
        fun symbolic(category: String, name: String, symbolicValue: Int): PredeclaredEnumerant {
            return PredeclaredEnumerant(
                category = category,
                name = name,
                value = LiteralExpr(symbolicValue, "i32", Span.INVALID),
                span = Span.INVALID
            )
        }
    }
}

/**
 * Registry for all predeclared enumerants defined in the WGSL specification.
 */
object PredeclaredEnumerantRegistry {
    private val _enumerants = mutableMapOf<String, List<PredeclaredEnumerant>>()
    
    // Initialize with all WGSL predeclared enumerants
    init {
        // AddressMode
        register(
            "AddressMode",
            listOf(
                PredeclaredEnumerant.symbolic("AddressMode", "clamp_to_edge", 0),
                PredeclaredEnumerant.symbolic("AddressMode", "repeat", 1),
                PredeclaredEnumerant.symbolic("AddressMode", "mirror_repeat", 2),
            )
        )
        
        // FilterMode
        register(
            "FilterMode",
            listOf(
                PredeclaredEnumerant.symbolic("FilterMode", "nearest", 0),
                PredeclaredEnumerant.symbolic("FilterMode", "linear", 1),
            )
        )
        
        // MipmapFilterMode
        register(
            "MipmapFilterMode",
            listOf(
                PredeclaredEnumerant.symbolic("MipmapFilterMode", "nearest", 0),
                PredeclaredEnumerant.symbolic("MipmapFilterMode", "linear", 1),
            )
        )
        
        // InterpolationType
        register(
            "InterpolationType",
            listOf(
                PredeclaredEnumerant.symbolic("InterpolationType", "perspective", 0),
                PredeclaredEnumerant.symbolic("InterpolationType", "linear", 1),
                PredeclaredEnumerant.symbolic("InterpolationType", "flat", 2),
            )
        )
        
        // InterpolationSampling
        register(
            "InterpolationSampling",
            listOf(
                PredeclaredEnumerant.symbolic("InterpolationSampling", "sample", 0),
                PredeclaredEnumerant.symbolic("InterpolationSampling", "vertex", 1),
                PredeclaredEnumerant.symbolic("InterpolationSampling", "centroid", 2),
            )
        )
        
        // ... autres catégories selon la spec
    }
    
    private fun register(category: String, enumerants: List<PredeclaredEnumerant>) {
        _enumerants[category] = enumerants
    }
    
    fun getEnumerant(category: String, name: String): PredeclaredEnumerant? {
        return _enumerants[category]?.find { it.name == name }
    }
    
    fun getAllEnumerants(): List<PredeclaredEnumerant> {
        return _enumerants.values.flatten()
    }
    
    fun getEnumerantsByCategory(category: String): List<PredeclaredEnumerant> {
        return _enumerants[category] ?: emptyList()
    }
    
    fun isKnownCategory(category: String): Boolean {
        return _enumerants.containsKey(category)
    }
    
    fun getValidValues(category: String): List<String> {
        return _enumerants[category]?.map { it.name } ?: emptyList()
    }
    
    fun isValidQualifiedName(qualifiedName: String): Boolean {
        val parts = qualifiedName.split(".")
        if (parts.size != 2) return false
        val category = parts[0]
        val name = parts[1]
        return getEnumerant(category, name) != null
    }
}
```

### 2. Expression.kt - EnumExpr unifiée

```kotlin
// Expression.kt - Modifications

/**
 * Common base class for all enum value references.
 * This unifies user-defined enum members and predeclared enumerants.
 */
sealed class EnumValueExpr : Expression() {
    abstract val enumValue: EnumValue
    
    /**
     * Returns the category/type name of this enum value.
     */
    abstract fun getEnumTypeName(): String
    
    /**
     * Returns the qualified name of this enum value.
     */
    fun getQualifiedName(): String = enumValue.getQualifiedName()
}

/**
 * Reference to a user-defined enum member.
 */
data class EnumMemberExpr(
    val enumDecl: EnumDecl,
    val memberName: String,
    override val span: Span,
) : EnumValueExpr() {
    override val enumValue: EnumValue by lazy {
        enumDecl.members.find { it.name == memberName }
            ?: throw IllegalStateException("Member $memberName not found in ${enumDecl.name}")
    }
    
    override fun getEnumTypeName(): String = enumDecl.name
    
    override fun <R> accept(visitor: ExpressionVisitor<R>): R {
        return visitor.visitEnumMember(this)
    }
}

/**
 * Reference to a predeclared enumerant value.
 */
data class PredeclaredEnumerantExpr(
    val category: String,
    val enumerant: PredeclaredEnumerant,
    override val span: Span,
) : EnumValueExpr() {
    override val enumValue: EnumValue = enumerant
    
    override fun getEnumTypeName(): String = category
    
    override fun <R> accept(visitor: ExpressionVisitor<R>): R {
        return visitor.visitPredeclaredEnumerant(this)
    }
}

// Mettre à jour ExpressionVisitor
interface ExpressionVisitor<R> {
    // ... méthodes existantes
    
    /**
     * Visits an enum member expression (user-defined).
     */
    fun visitEnumMember(expr: EnumMemberExpr): R
    
    /**
     * Visits a predeclared enumerant expression.
     */
    fun visitPredeclaredEnumerant(expr: PredeclaredEnumerantExpr): R
}

// Mettre à jour DefaultExpressionVisitor
class DefaultExpressionVisitor<R> : ExpressionVisitor<R> {
    // ... implémentations existantes
    
    override fun visitEnumMember(expr: EnumMemberExpr): R = defaultResult()
    override fun visitPredeclaredEnumerant(expr: PredeclaredEnumerantExpr): R = defaultResult()
}
```

### 3. TypeResolver.kt - Intégration unifiée

```kotlin
// TypeResolver.kt - Modifications

/**
 * Type resolver that handles both user-defined and predeclared enums.
 */
class TypeResolver(private val translationUnit: TranslationUnit) : StatementVisitor<Unit>, ExpressionVisitor<Type> {
    // ... code existant
    
    override fun visitEnumMember(expr: EnumMemberExpr): Type {
        // Résoudre le type de l'enum déclarée
        return resolveEnumType(expr.enumDecl)
    }
    
    override fun visitPredeclaredEnumerant(expr: PredeclaredEnumerantExpr): Type {
        // Résoudre le type de l'enum prédéfini
        return resolvePredeclaredEnumType(expr.category)
    }
    
    /**
     * Resolves the type of a user-defined enum.
     */
    private fun resolveEnumType(enumDecl: EnumDecl): Type {
        return Type.EnumType(enumDecl.name, enumDecl)
    }
    
    /**
     * Resolves the type of a predeclared enum category.
     */
    private fun resolvePredeclaredEnumType(category: String): Type {
        if (!PredeclaredEnumerantRegistry.isKnownCategory(category)) {
            throw TypeResolutionError("Unknown enum category: $category")
        }
        return Type.PredeclaredEnumType(category)
    }
    
    /**
     * Checks if a type is an enum type (user-defined or predeclared).
     */
    fun isEnumType(type: Type): Boolean {
        return when (type) {
            is Type.EnumType -> true
            is Type.PredeclaredEnumType -> true
            else -> false
        }
    }
    
    /**
     * Gets all enum values for a given enum type.
     */
    fun getEnumValues(type: Type): List<EnumValue> {
        return when (type) {
            is Type.EnumType -> {
                val enumDecl = translationUnit.findEnum(type.name)
                enumDecl?.members ?: emptyList()
            }
            is Type.PredeclaredEnumType -> {
                PredeclaredEnumerantRegistry.getEnumerantsByCategory(type.category)
            }
            else -> emptyList()
        }
    }
}

// Extension pour trouver les déclarations d'enum
fun TranslationUnit.findEnum(name: String): EnumDecl? {
    return declarations
        .filterIsInstance<EnumDecl>()
        .find { it.name == name }
}
```

### 4. Parser.kt - Parsing unifié

```kotlin
// Parser.kt - Modifications pour parsing unifié

/**
 * Parses a member access that could be either:
 * - A regular member access (x.y)
 * - A user-defined enum member (MyEnum.VALUE)
 * - A predeclared enumerant (AddressMode.clamp_to_edge)
 */
private fun parseMemberAccess(base: Expression): Expression {
    val start = base.span
    expect(TokenKind.DOT)
    val memberName = expectIdentifier()
    
    // Cas 1: Predeclared enumerant
    if (base is IdentExpr) {
        val category = base.identifier
        val enumerant = PredeclaredEnumerantRegistry.getEnumerant(category, memberName)
        if (enumerant != null) {
            return PredeclaredEnumerantExpr(category, enumerant, spanFrom(start))
        }
    }
    
    // Cas 2: User-defined enum member (à résoudre plus tard)
    // Pour l'instant, créer un MemberAccessExpr qui sera résolu par le TypeResolver
    return MemberAccessExpr(base, memberName, spanFrom(start))
}

/**
 * After parsing, resolve member accesses to enum members if appropriate.
 * This is called during semantic analysis.
 */
private fun resolveMemberAccesses(unit: TranslationUnit): TranslationUnit {
    val resolver = MemberAccessResolver(unit)
    return resolver.resolve()
}

/**
 * Resolves MemberAccessExpr to EnumMemberExpr when the base is an enum type.
 */
class MemberAccessResolver(private val unit: TranslationUnit) : ExpressionTransformer() {
    private val enumDecls: Map<String, EnumDecl> by lazy {
        unit.declarations
            .filterIsInstance<EnumDecl>()
            .associateBy { it.name }
    }
    
    override fun transform(expr: MemberAccessExpr): Expression {
        // Vérifier si la base est une référence à un enum user-defined
        if (expr.base is IdentExpr) {
            val enumName = expr.base.identifier
            val enumDecl = enumDecls[enumName]
            
            if (enumDecl != null) {
                // C'est une référence à un membre d'enum user-defined
                return EnumMemberExpr(enumDecl, expr.member, expr.span)
            }
        }
        
        return expr
    }
}
```

```kotlin
// EnumIntegrationTest.kt
package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.types.shouldBeInstanceOf
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.lexer.Lexer

/**
 * Tests for integration between user-defined enums and predeclared enumerants.
 */
class EnumIntegrationTest : FunSpec({
    
    context("user-defined and predeclared enums in same context") {
        test("both types can be used together") {
            val source = """
                enum MyEnum { VALUE1, VALUE2 }
                let userEnum = MyEnum.VALUE1;
                let preEnum = AddressMode.clamp_to_edge;
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("same access syntax for both") {
            val source = """
                enum Color { RED, GREEN, BLUE }
                let c1 = Color.RED;
                let c2 = AddressMode.repeat;
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("type resolution") {
        test("user-defined enum type resolution") {
            val source = """
                enum Status { ACTIVE, INACTIVE }
                let s: Status = Status.ACTIVE;
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            val resolver = TypeResolver(unit)
            
            // Vérifier que Status est résolu comme un type enum
            parser.errors.isEmpty() shouldBe true
        }
        
        test("predeclared enum type resolution") {
            val source = "let mode: AddressMode = AddressMode.clamp_to_edge;"
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            val resolver = TypeResolver(unit)
            
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("expression types") {
        test("EnumMemberExpr is created for user-defined enums") {
            val source = """
                enum Direction { LEFT, RIGHT }
                let d = Direction.LEFT;
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
            
            // Après résolution, on devrait avoir un EnumMemberExpr
            // (ceci sera vérifié via un visitor)
        }
        
        test("PredeclaredEnumerantExpr is created for predeclared enums") {
            val source = "let mode = AddressMode.clamp_to_edge;"
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            // Devrait contenir un PredeclaredEnumerantExpr
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("function parameters") {
        test("user-defined enum in function parameters") {
            val source = """
                enum Mode { FAST, SLOW }
                fn process(mode: Mode) {}
                fn main() { process(Mode.FAST); }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
        
        test("predeclared enum in function parameters") {
            val source = """
                fn sample(mode: AddressMode) {}
                fn main() { sample(AddressMode.clamp_to_edge); }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("mixed usage") {
        test("complex example with both enum types") {
            val source = """
                enum TextureType { DIFFUSE, NORMAL, SPECULAR }
                
                struct Material {
                    type: TextureType,
                    addressMode: AddressMode,
                    filterMode: FilterMode,
                }
                
                let mat = Material(
                    TextureType.DIFFUSE,
                    AddressMode.repeat,
                    FilterMode.linear
                );
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parseAndValidate()
            
            parser.errors.isEmpty() shouldBe true
        }
    }
    
    context("enum value properties") {
        test("getQualifiedName works for both enum types") {
            // Ce test vérifie que les deux types d'enum ont des qualified names
            val userEnumValue = EnumMember("VALUE1", null, emptyList(), Span.INVALID)
            val preEnumValue = PredeclaredEnumerant("AddressMode", "clamp_to_edge", null)
            
            userEnumValue.getQualifiedName() shouldBe "VALUE1"
            preEnumValue.getQualifiedName() shouldBe "AddressMode.clamp_to_edge"
        }
        
        test("EnumValueExpr common interface") {
            // Vérifier que les deux implémentent EnumValueExpr
            val userExpr = EnumMemberExpr(
                EnumDecl("MyEnum", emptyList(), emptyList(), Span.INVALID),
                "VALUE1",
                Span.INVALID
            )
            val preExpr = PredeclaredEnumerantExpr(
                "AddressMode",
                PredeclaredEnumerant("AddressMode", "clamp_to_edge", null),
                Span.INVALID
            )
            
            (userExpr is EnumValueExpr) shouldBe true
            (preExpr is EnumValueExpr) shouldBe true
        }
    }
})
```
