// =============================================================================
// ABSTRACT TYPE DECLARATIONS

/**
 * Abstract integer type declaration.
 * 
 * This represents the `abstract int` type in WGSL, which is an integer type
 * without a specific underlying representation.
 * 
 * @property span The source span of this type declaration.
 */
data class AbstractIntType(
    override val span: Span,
) : TypeDecl() {
    /**
     * Returns a string representation of this type.
     */
    override fun toString(): String = "abstract int"
    
    /**
     * Checks if this type is equal to another type.
     */
    override fun equals(other: Any?): Boolean {
        return other is AbstractIntType
    }
    
    /**
     * Returns the hash code for this type.
     */
    override fun hashCode(): Int = AbstractIntType::class.hashCode()
}

/**
 * Abstract float type declaration.
 * 
 * This represents the `abstract float` type in WGSL, which is a floating-point
 * type without a specific underlying representation.
 * 
 * @property span The source span of this type declaration.
 */

    override val span: Span,

    /**
     * Returns a string representation of this type.
     */
    override fun toString(): String = "abstract float"

     * Checks if this type is equal to another type.

 * Visitor interface for type declarations.

 * Default implementation of TypeDeclVisitor.
 */
open class DefaultTypeDeclVisitor<R> : TypeDeclVisitor<R> {
    protected open fun defaultResult(): R {
        @Suppress("UNCHECKED_CAST")
        return null as R

    // ... implémentations existantes

// AbstractTypeAstTest.kt
package io.ygdrasil.wgsl.ast

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldNotBe
import io.ygdrasil.wgsl.ir.Span

/**
 * Tests for abstract type AST classes.
 */
class AbstractTypeAstTest : FunSpec({
    
    context("AbstractIntType") {
        test("create AbstractIntType") {
            val span = Span(0u, 12u)
            val type = AbstractIntType(span)
            
            type.span shouldBe span
        }
        
        test("AbstractIntType toString") {
            val type = AbstractIntType(Span(0u, 12u))
            type.toString() shouldBe "abstract int"
        }
        
        test("AbstractIntType equality") {
            val type1 = AbstractIntType(Span(0u, 12u))
            val type2 = AbstractIntType(Span(10u, 22u))
            
            type1 shouldBe type2
            type1.hashCode() shouldBe type2.hashCode()
        }
        
        test("AbstractIntType inequality with AbstractFloatType") {
            val intType = AbstractIntType(Span(0u, 12u))
            val floatType = AbstractFloatType(Span(0u, 14u))
            
            intType shouldNotBe floatType
        }
        
        test("AbstractIntType visitor") {
            val type = AbstractIntType(Span(0u, 12u))
            val visitor = object : TypeDeclVisitor<String> {
                override fun visitAbstractIntType(type: AbstractIntType): String = "abstract int"
                override fun visitAbstractFloatType(type: AbstractFloatType): String = "abstract float"
                // ... autres méthodes
            }
            
            type.accept(visitor) shouldBe "abstract int"
        }
    }
    
    context("AbstractFloatType") {
        test("create AbstractFloatType") {
            val span = Span(0u, 14u)
            val type = AbstractFloatType(span)
            
            type.span shouldBe span
        }
        
        test("AbstractFloatType toString") {
            val type = AbstractFloatType(Span(0u, 14u))
            type.toString() shouldBe "abstract float"
        }
        
        test("AbstractFloatType equality") {
            val type1 = AbstractFloatType(Span(0u, 14u))
            val type2 = AbstractFloatType(Span(10u, 24u))
            
            type1 shouldBe type2
            type1.hashCode() shouldBe type2.hashCode()
        }
        
        test("AbstractFloatType visitor") {
            val type = AbstractFloatType(Span(0u, 14u))
            val visitor = object : TypeDeclVisitor<String> {
                override fun visitAbstractIntType(type: AbstractIntType): String = "abstract int"
                override fun visitAbstractFloatType(type: AbstractFloatType): String = "abstract float"
                // ... autres méthodes
            }
            
            type.accept(visitor) shouldBe "abstract float"
        }
    }
    
    context("Type discrimination") {
        test("AbstractIntType is TypeDecl") {
            val type = AbstractIntType(Span(0u, 12u))
            type shouldBeInstanceOf<TypeDecl>()
        }
        
        test("AbstractFloatType is TypeDecl") {
            val type = AbstractFloatType(Span(0u, 14u))
            type shouldBeInstanceOf<TypeDecl>()
        }
    }
})
