// =============================================================================
// ABSTRACT TYPE RESOLUTION
// =============================================================================

/**
 * Represents the abstract int type in the type system.
 */
data class AbstractIntTypeInner(
    override val span: Span,
) : TypeInner.Abstract()

/**
 * Represents the abstract float type in the type system.
 */
data class AbstractFloatTypeInner(
    override val span: Span,
) : TypeInner.Abstract()

// Dans TypeResolver class

/**
 * Resolves an AbstractIntType declaration to a Type.
 */
private fun resolveAbstractIntType(typeDecl: AbstractIntType): Type {
    return Type(
        TypeInner.AbstractInt(typeDecl.span),
        typeDecl.span
    )
}

/**
 * Resolves an AbstractFloatType declaration to a Type.
 */
private fun resolveAbstractFloatType(typeDecl: AbstractFloatType): Type {
    return Type(
        TypeInner.AbstractFloat(typeDecl.span),
        typeDecl.span
    )
}

// Mettre à jour la méthode principale resolveType
private fun resolveType(decl: TypeDecl): Type? {
    return when (decl) {
        is AbstractIntType -> resolveAbstractIntType(decl)
        is AbstractFloatType -> resolveAbstractFloatType(decl)
        // ... autres cas existants
        else -> null
    }
}

// Ajouter des propriétés pour les types abstraits
val Type.isAbstractInt: Boolean
    get() = this.inner is TypeInner.AbstractInt

val Type.isAbstractFloat: Boolean
    get() = this.inner is TypeInner.AbstractFloat

val Type.isAbstract: Boolean
    get() = this.inner is TypeInner.Abstract

// Ajouter des méthodes de conversion
fun Type.toAbstractInt(): Type? {
    if (this.isAbstractInt) return this
    // Conversion possible depuis les types int concrets
    if (this.inner is TypeInner.Scalar && this.inner.width == 32 && this.inner.kind == ScalarKind.I32) {
        return Type(TypeInner.AbstractInt(this.span), this.span)
    }
    return null
}

fun Type.toAbstractFloat(): Type? {
    if (this.isAbstractFloat) return this
    // Conversion possible depuis les types float concrets
    if (this.inner is TypeInner.Scalar && this.inner.width == 32 && this.inner.kind == ScalarKind.F32) {
        return Type(TypeInner.AbstractFloat(this.span), this.span)
    }
    return null
}

// AbstractTypeResolverTest.kt
package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.shouldBe
import io.kotest.matchers.shouldNotBe
import io.kotest.matchers.types.shouldBeInstanceOf
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.ir.Span

/**
 * Tests for abstract type resolution.
 */
class AbstractTypeResolverTest : FunSpec({
    
    lateinit var typeResolver: TypeResolver
    
    beforeTest {
        // Créer un TypeResolver avec des types pour tests
        typeResolver = TypeResolver(emptyList())
    }
    
    context("AbstractIntType resolution") {
        test("resolve AbstractIntType to Type") {
            val typeDecl = AbstractIntType(Span(0u, 12u))
            val resolvedType = typeResolver.resolveType(typeDecl)
            
            resolvedType shouldNotBe null
            resolvedType?.inner shouldBeInstanceOf<TypeInner.AbstractInt>()
        }
        
        test("AbstractIntType is abstract") {
            val typeDecl = AbstractIntType(Span(0u, 12u))
            val resolvedType = typeResolver.resolveType(typeDecl)!!
            
            resolvedType.isAbstract shouldBe true
            resolvedType.isAbstractInt shouldBe true
            resolvedType.isAbstractFloat shouldBe false
        }
        
        test("convert concrete int to abstract int") {
            // Créer un type i32 concret
            val scalarType = ScalarType(ScalarKind.I32, 32u, Span(0u, 4u))
            val concreteType = Type(TypeInner.Scalar(ScalarKind.I32, 32u), Span(0u, 4u))
            
            val abstractType = concreteType.toAbstractInt()
            abstractType shouldNotBe null
            abstractType?.isAbstractInt shouldBe true
        }
    }
    
    context("AbstractFloatType resolution") {
        test("resolve AbstractFloatType to Type") {
            val typeDecl = AbstractFloatType(Span(0u, 14u))
            val resolvedType = typeResolver.resolveType(typeDecl)
            
            resolvedType shouldNotBe null
            resolvedType?.inner shouldBeInstanceOf<TypeInner.AbstractFloat>()
        }
        
        test("AbstractFloatType is abstract") {
            val typeDecl = AbstractFloatType(Span(0u, 14u))
            val resolvedType = typeResolver.resolveType(typeDecl)!!
            
            resolvedType.isAbstract shouldBe true
            resolvedType.isAbstractFloat shouldBe true
            resolvedType.isAbstractInt shouldBe false
        }
        
        test("convert concrete float to abstract float") {
            // Créer un type f32 concret
            val concreteType = Type(TypeInner.Scalar(ScalarKind.F32, 32u), Span(0u, 4u))
            
            val abstractType = concreteType.toAbstractFloat()
            abstractType shouldNotBe null
            abstractType?.isAbstractFloat shouldBe true
        }
    }
    
    context("Type discrimination") {
        test("AbstractIntType vs AbstractFloatType") {
            val intType = typeResolver.resolveType(AbstractIntType(Span(0u, 12u)))!!
            val floatType = typeResolver.resolveType(AbstractFloatType(Span(0u, 14u)))!!
            
            intType.isAbstractInt shouldBe true
            intType.isAbstractFloat shouldBe false
            
            floatType.isAbstractInt shouldBe false
            floatType.isAbstractFloat shouldBe true
        }
        
        test("Abstract types are not concrete") {
            val abstractInt = typeResolver.resolveType(AbstractIntType(Span(0u, 12u)))!!
            val concreteInt = Type(TypeInner.Scalar(ScalarKind.I32, 32u), Span(0u, 4u))
            
            abstractInt.inner shouldNotBeInstanceOf<TypeInner.Scalar>()
            concreteInt.inner shouldBeInstanceOf<TypeInner.Scalar>()
        }
    }
})
