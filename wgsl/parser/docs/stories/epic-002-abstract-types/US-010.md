## 🎯 Suite de Tests Complète

```kotlin
// AbstractTypeAllTests.kt
package io.ygdrasil.wgsl.parser

import io.kotest.core.spec.style.FunSpec
import io.ygdrasil.wgsl.ast.*
import io.ygdrasil.wgsl.lexer.Lexer
import io.ygdrasil.wgsl.ir.Span

/**
 * Comprehensive test suite for abstract numeric types support.
 * 
 * This test suite combines all abstract type tests to ensure:
 * - Lexing works correctly
 * - Parsing works correctly
 * - Type resolution works correctly
 * - Integration with the rest of the parser works correctly
 */
class AbstractTypeAllTests : FunSpec({
    
    // =========================================================================
    // LEXING TESTS
    // =========================================================================
    
    context("Lexing: abstract type keywords") {
        test("recognizes abstract keyword") {
            val tokens = Lexer("abstract").tokenizeSignificant()
            tokens shouldHaveSize 1
            tokens[0].kind shouldBe TokenKind.ABSTRACT
        }
        
        test("recognizes int keyword") {
            val tokens = Lexer("int").tokenizeSignificant()
            tokens shouldHaveSize 1
            tokens[0].kind shouldBe TokenKind.INT
        }
        
        test("recognizes float keyword") {
            val tokens = Lexer("float").tokenizeSignificant()
            tokens shouldHaveSize 1
            tokens[0].kind shouldBe TokenKind.FLOAT
        }
        
        test("recognizes abstract int") {
            val tokens = Lexer("abstract int").tokenizeSignificant()
            tokens shouldHaveSize 2
            tokens[0].kind shouldBe TokenKind.ABSTRACT
            tokens[1].kind shouldBe TokenKind.INT
        }
        
        test("recognizes abstract float") {
            val tokens = Lexer("abstract float").tokenizeSignificant()
            tokens shouldHaveSize 2
            tokens[0].kind shouldBe TokenKind.ABSTRACT
            tokens[1].kind shouldBe TokenKind.FLOAT
        }
    }
    
    // =========================================================================
    // PARSING TESTS
    // =========================================================================
    
    context("Parsing: abstract types in declarations") {
        test("parse abstract int in variable declaration") {
            val source = "let x: abstract int;"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            unit.declarations shouldHaveSize 1
            val varDecl = unit.declarations[0] as VariableDeclStatement
            varDecl.decl.type shouldBeInstanceOf<AbstractIntType>()
            parser.errors.isEmpty() shouldBe true
        }
        
        test("parse abstract float in variable declaration") {
            val source = "let y: abstract float;"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val varDecl = unit.declarations[0] as VariableDeclStatement
            varDecl.decl.type shouldBeInstanceOf<AbstractFloatType>()
        }
        
        test("parse abstract int with initializer") {
            val source = "let x: abstract int = 42;"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val varDecl = unit.declarations[0] as VariableDeclStatement
            varDecl.decl.type shouldBeInstanceOf<AbstractIntType>()
            varDecl.decl.initializer shouldNotBe null
        }
        
        test("parse abstract int as function parameter") {
            val source = "fn process(x: abstract int) { }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val funcDecl = unit.declarations[0] as FunctionDecl
            funcDecl.parameters shouldHaveSize 1
            funcDecl.parameters[0].type shouldBeInstanceOf<AbstractIntType>()
        }
        
        test("parse abstract float as function return type") {
            val source = "fn getValue() -> abstract float { return 3.14; }"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val funcDecl = unit.declarations[0] as FunctionDecl
            funcDecl.returnType shouldBeInstanceOf<AbstractFloatType>()
        }
        
        test("parse abstract types in struct") {
            val source = """
                struct Config {
                    intValue: abstract int,
                    floatValue: abstract float,
                }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            val structDecl = unit.declarations[0] as StructDecl
            structDecl.members shouldHaveSize 2
            structDecl.members[0].type shouldBeInstanceOf<AbstractIntType>()
            structDecl.members[1].type shouldBeInstanceOf<AbstractFloatType>()
        }
    }
    
    // =========================================================================
    // TYPE RESOLUTION TESTS
    // =========================================================================
    
    context("Type Resolution: abstract types") {
        lateinit var typeResolver: TypeResolver
        
        beforeTest {
            typeResolver = TypeResolver(emptyList())
        }
        
        test("resolve AbstractIntType to Type") {
            val typeDecl = AbstractIntType(Span(0u, 12u))
            val resolvedType = typeResolver.resolveType(typeDecl)
            
            resolvedType shouldNotBe null
            resolvedType?.isAbstractInt shouldBe true
        }
        
        test("resolve AbstractFloatType to Type") {
            val typeDecl = AbstractFloatType(Span(0u, 14u))
            val resolvedType = typeResolver.resolveType(typeDecl)
            
            resolvedType shouldNotBe null
            resolvedType?.isAbstractFloat shouldBe true
        }
        
        test("AbstractIntType and AbstractFloatType are different") {
            val intType = typeResolver.resolveType(AbstractIntType(Span(0u, 12u)))!!
            val floatType = typeResolver.resolveType(AbstractFloatType(Span(0u, 14u)))!!
            
            intType.isAbstractInt shouldBe true
            intType.isAbstractFloat shouldBe false
            
            floatType.isAbstractInt shouldBe false
            floatType.isAbstractFloat shouldBe true
        }
    }
    
    // =========================================================================
    // AST TESTS
    // =========================================================================
    
    context("AST: abstract type classes") {
        test("create AbstractIntType") {
            val span = Span(0u, 12u)
            val type = AbstractIntType(span)
            
            type.span shouldBe span
            type.toString() shouldBe "abstract int"
        }
        
        test("create AbstractFloatType") {
            val span = Span(0u, 14u)
            val type = AbstractFloatType(span)
            
            type.span shouldBe span
            type.toString() shouldBe "abstract float"
        }
        
        test("AbstractIntType equality") {
            val type1 = AbstractIntType(Span(0u, 12u))
            val type2 = AbstractIntType(Span(10u, 22u))
            
            type1 shouldBe type2
            type1.hashCode() shouldBe type2.hashCode()
        }
        
        test("AbstractFloatType equality") {
            val type1 = AbstractFloatType(Span(0u, 14u))
            val type2 = AbstractFloatType(Span(10u, 24u))
            
            type1 shouldBe type2
            type1.hashCode() shouldBe type2.hashCode()
        }
        
        test("AbstractIntType and AbstractFloatType are not equal") {
            val intType = AbstractIntType(Span(0u, 12u))
            val floatType = AbstractFloatType(Span(0u, 14u))
            
            intType shouldNotBe floatType
        }
    }
    
    // =========================================================================
    // INTEGRATION TESTS
    // =========================================================================
    
    context("Integration: abstract types in complete modules") {
        test("parse complete module with abstract types") {
            val source = """
                let a: abstract int = 10;
                let b: abstract float = 2.5;
                
                struct Data {
                    x: abstract int,
                    y: abstract float,
                }
                
                fn convert(x: abstract int) -> abstract float {
                    return abstract float(x);
                }
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            parser.errors.isEmpty() shouldBe true
            unit.declarations shouldHaveSize 4
        }
        
        test("parse abstract types with concrete types") {
            val source = """
                let a: i32 = 10;
                let b: abstract int = a;
                let c: f32 = 3.14;
                let d: abstract float = c;
            """.trimIndent()
            
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            parser.errors.isEmpty() shouldBe true
            unit.declarations shouldHaveSize 4
        }
    }
    
    // =========================================================================
    // ERROR HANDLING TESTS
    // =========================================================================
    
    context("Error Handling: invalid abstract type usage") {
        test("error on incomplete abstract type") {
            val source = "let x: abstract;"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            parser.errors.isNotEmpty() shouldBe true
        }
        
        test("error on invalid abstract type") {
            val source = "let x: abstract bool;"
            val parser = Parser(Lexer(source))
            val unit = parser.parse()
            
            parser.errors.isNotEmpty() shouldBe true
        }
    }
})
```
