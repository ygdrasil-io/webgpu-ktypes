## 🧪 Tests Attendus

```kotlin
// AbstractTypeLexerTest.kt
package io.ygdrasil.wgsl.lexer

import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.collections.shouldContain
import io.kotest.matchers.collections.shouldHaveSize
import io.kotest.matchers.shouldBe

/**
 * Tests for lexing abstract numeric type keywords.
 */
class AbstractTypeLexerTest : FunSpec({
    
    context("abstract keyword") {
        test("recognizes abstract as keyword") {
            val tokens = Lexer("abstract").tokenizeSignificant()
            tokens shouldHaveSize 1
            tokens[0].kind shouldBe TokenKind.ABSTRACT
        }
        
        test("abstract is a keyword") {
            TokenKind.ABSTRACT.isKeyword shouldBe true
        }
    }
    
    context("int keyword") {
        test("recognizes int as keyword") {
            val tokens = Lexer("int").tokenizeSignificant()
            tokens shouldHaveSize 1
            tokens[0].kind shouldBe TokenKind.INT
        }
        
        test("int is a keyword") {
            TokenKind.INT.isKeyword shouldBe true
        }
    }
    
    context("float keyword") {
        test("recognizes float as keyword") {
            val tokens = Lexer("float").tokenizeSignificant()
            tokens shouldHaveSize 1
            tokens[0].kind shouldBe TokenKind.FLOAT
        }
        
        test("float is a keyword") {
            TokenKind.FLOAT.isKeyword shouldBe true
        }
    }
    
    context("abstract numeric types") {
        test("abstract int") {
            val tokens = Lexer("abstract int").tokenizeSignificant()
            tokens shouldHaveSize 2
            tokens[0].kind shouldBe TokenKind.ABSTRACT
            tokens[1].kind shouldBe TokenKind.INT
        }
        
        test("abstract float") {
            val tokens = Lexer("abstract float").tokenizeSignificant()
            tokens shouldHaveSize 2
            tokens[0].kind shouldBe TokenKind.ABSTRACT
            tokens[1].kind shouldBe TokenKind.FLOAT
        }
    }
    
    context("in type declarations") {
        test("abstract int in variable declaration") {
            val source = "let x: abstract int;"
            val tokens = Lexer(source).tokenizeSignificant()
            
            tokens shouldContain TokenKind.LET
            tokens shouldContain TokenKind.COLON
            tokens shouldContain TokenKind.ABSTRACT
            tokens shouldContain TokenKind.INT
            tokens shouldContain TokenKind.SEMICOLON
        }
    }
})
```

## 📊 Conformité avec la Spécification WGSL

### Section 6.2.1 - Abstract Numeric Types

La spécification WGSL définit deux types numériques abstraits :

> **abstract int** - Represents an abstract integer value without specifying the underlying representation
> **abstract float** - Represents an abstract floating-point value without specifying the underlying representation

**Syntaxe** :
```wgsl
let x: abstract int = 42;
let y: abstract float = 3.14;

fn process(x: abstract int) -> abstract float {
    return abstract float(x);
}
```
